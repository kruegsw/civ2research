```
if has_tech(ownerCiv, TRADE=0x25):  // advance 37

if not has_tech(ownerCiv, CONSTRUCTION=0x41):

if has_building(citySlot, LIBRARY=9):

if continent is odd and continent > 1:

// --- Supply[7]: Spice (commodity 7) ---

rawGems = riverCount*5 - terrainCount[2]

rawGems = clamp(rawGems, 0, rawForest)
rawForest = clamp(rawForest, 0, rawGems)

    supply[8] <<= 1

if has_tech(ownerCiv, TRADE=0x25): demand[1] <<= 1

    weight /= 2
    remainingSize -= portion

if city.size < 8: demand[5] >>= 1

if has_tech(ownerCiv, MAGNETISM=0x17): demand[5] <<= 1
if has_building(citySlot, FACTORY=19): demand[5] <<= 1

if has_building(citySlot, MARKETPLACE=5): demand[5] = demand[5] * 3 / 2
if has_building(citySlot, BANK=10): demand[5] = demand[5] * 3 / 2
if has_tech(ownerCiv, MAGNETISM=0x17): demand[5] = demand[5] * 3 / 2
if has_tech(ownerCiv, NAVIGATION=0x10): demand[5] = (demand[5] + 3) / 4
if city.size < 5: demand[5] >>= 1

// --- Demand[6]: Wine demand (Copper if roadCount involved) ---
demand[6] = supply[3] + roadCount  // NOTE: roadCount UNINITIALIZED

// --- Demand[7]: Spice demand ---
// (city.x + city.y) % 3 determines which of demand slots 10/12/13 gets city.size*8
// The remaining two get 0 each. One gets the wine/spice/gold demand

hash = (city.x + city.y) % 3
if hash == 1: baseSlot = 12; demand[12] = city.size * 8
elif hash == 2: baseSlot = 13; demand[13] = city.size * 8

// --- Demand[baseSlot]: location-based demand ---
// Boosted by Marketplace(0x0b)+Writing tech, or Bank(0x0a)
if has_building(citySlot, MARKETPLACE=11) or has_tech(ownerCiv, WRITING=10):
    demand[baseSlot] = demand[baseSlot] * 3 / 2
if has_building(citySlot, BANK=10):
    demand[baseSlot] = demand[baseSlot] * 3 / 2
if has_tech(ownerCiv, MATHEMATICS=0x16): demand[baseSlot] >>= 1
if has_tech(ownerCiv, NAVIGATION=0x10): demand[baseSlot] >>= 1
if civData[ownerCiv].leaderIndex == 17: demand[baseSlot] <<= 1

// --- Demand[10]: Ivory/Resources demand ---
aridity = scienceRate - 12
if aridity < 0: aridity = 0
demand[10] = terrain_food_yield[continent]/10 - aridity
if city.size < 4: demand[10] /= 2
if terrain_food_yield[continent] > 400:
    if city.size > 7: demand[10] <<= 1
    if -(city.x - city.y) & 3 == 0: demand[10] = 0
if has_tech(ownerCiv, MINIATURIZATION=0x46): demand[10] >>= 1

if not has_tech(ownerCiv, TRADE=0x25):
    demand[14] = -1  // unavailable
else:

    if demand[14] <= supply[14]: demand[14] >>= 1
    if not has_building(citySlot, NUCLEAR_PLANT=15):

        if city.size < 10: demand[14] >>= 1
        if city.size < 20: demand[14] >>= 1

    if has_building(citySlot, SDI_DEFENSE=25): demand[14] <<= 1
    if has_building(citySlot, SAM_BATTERY=13): demand[14] >>= 1
    if has_building(citySlot, COASTAL_FORTRESS=18): demand[14] >>= 1

    demand[15] = -1  // unavailable
else:
    demand[15] = scienceRate^2
    shift15 = clamp(sizeTier - 3, -3, 0)
    demand[15] = arithmetic_shift(demand[15], shift15)
    if has_building(citySlot, POWER_PLANT=21) or has_building(citySlot, SAM_BATTERY=17):
        demand[15] <<= 1

demand[8] = sizeTier * 4 + 4 + abs(city.x - city.y)

// --- Demand[9]: Gold demand ---
demand[9] = (mapHeight/2 - distY)*2 - distX + mapWidth/2 +
            terrainCount[9]*4 + terrainCount[1]/2 + terrainCount[8]*2 + terrainCount[0]*4

demand[9] = arithmetic_shift(demand[9], shift9)
if continent == 1 and (citySlot & 2) != 0:
    demand[9] = demand[9] * 3 / 2
if city.size < 7: demand[9] >>= 1

// --- Demand[11]: Silk demand ---
demand[11] = city.size * 8

// --- Demand[12/13]: (set above by hash) ---

// === SUPPLY vs DEMAND RESOLUTION ===
// For each commodity: if supply > demand, zero out the smaller one
// (commodity can be supplied OR demanded, not both)
for i = 0..15:
    if supply[i] < demand[i]:
        if supply[i] > 0: supply[i] = 0
    elif demand[i] > 0:
        demand[i] = 0

// === SORT AND SELECT TOP 3 ===
// Sort supply indices by value (ascending) using bubble sort FUN_00414f02
sort(supply_indices[16], supply_values[16])
sort(demand_indices[16], demand_values[16])

// Pick top 3 supply commodities (highest values, skip negatives)
supplyIdx = 15  // start from end (highest after ascending sort)
for slot = 0..2:
    while supply_values[supplyIdx] < 0: supplyIdx--  // skip unavailable
    city.supply[slot] = supply_indices[supplyIdx]
    supplyIdx--

// Pick top 3 demand commodities (highest values, skip negatives)
demandIdx = 15
for slot = 0..2:
    while demand_values[demandIdx] < 0: demandIdx--
    city.demand[slot] = demand_indices[demandIdx]
    demandIdx--

// === EXTRA COMMODITY SLOTS (position-hash based) ===
// Two extra commodity slots determined by city position hash
if scienceRate < 32:
    extraSlot1 = (city.y*5 + city.x*3) % 14
    extraSlot2 = (city.y*7 + city.x*13) % 14
else:
    extraSlot1 = (city.y*5 + city.x*3) % 9 + 5
    extraSlot2 = (city.y*7 + city.x*13) % 9 + 5

if extraSlot1 == extraSlot2: extraSlot2 = -2  // invalid

// Check if already in supply or demand
flags = 0
for slot = 0..2:
    if city.demand[slot] > 13: flags |= 1
    if city.supply[slot] > 13: flags |= 2
    if city.supply[slot] == extraSlot1 or city.demand[slot] == extraSlot1: extraSlot1 = -2
    if city.supply[slot] == extraSlot2 or city.demand[slot] == extraSlot2: extraSlot2 = -2

// Assign uranium/fusion extras if unassigned
if extraSlot1 < 0 and not (flags & 1):
    if has_tech(TRADE) and extraSlot1 is even: extraSlot1 = 14  // uranium
    if has_tech(FUSION_POWER) and extraSlot1 is even: extraSlot1 = 15  // fusion
if extraSlot1 >= 0: city.extraDemand = extraSlot1

if extraSlot2 < 0 and not (flags & 2):
    extraSlot2 = (extraSlot2 & 1) + 14
    if extraSlot2 == 14 and not has_tech(TRADE): extraSlot2 = -1
    if extraSlot2 == 15 and not has_tech(FUSION_POWER): extraSlot2 = -1
if extraSlot2 >= 0: city.extraSupply = extraSlot2

// === MARK ACTIVE TRADE ROUTES ===
// Negate supply entries for commodities being actively traded
for each trade route in city:
    commodityId = tradeRoute.commodity
    if commodityId >= 0:
        for slot = 0..2:
            if city.supply[slot] == commodityId:
                city.supply[slot] = -city.supply[slot]  // negate = "in use"

        // Also negate demand in partner city
        partnerCity = tradeRoute.partnerCity
        partnerCommodity = find matching commodity
        if partnerCommodity >= 0:
            for slot = 0..2:
                if city.demand[slot] == partnerCommodity:
                    city.demand[slot] = -city.demand[slot]

// === MARK FREIGHT UNITS ===
// Check all freight/caravan units belonging to this civ
for each unit:
    if unit.alive and unit.owner == cityOwner and
       unit.type == FREIGHT(7) and unit.cargo >= 0:
        for slot = 0..2:
            if city.supply[slot] == unit.cargo:
                city.supply[slot] = -city.supply[slot]
```
- **Notes**: 8227 bytes — largest function in this block by far. Calculates 16 supply and 16 demand commodity values per city based on terrain composition, city size, distance from map center, continent ID, technology, buildings, and civ-specific bonuses. Key arrays: supply at DAT_0063f668[16], demand at DAT_0063f540[16]. Terrain types: 0=desert, 1=plains, 2=grassland, 3=forest, 4=hills, 5=mountains, 6=tundra+jungle(merged), 7=jungle(zeroed), 8=swamp, 9=arctic, 10=ocean.   **Recalc trigger**: `(city.attribs3 & 0x02) || globalRevealFlag || ((turn+citySlot) % 16 == 0)`. NOT every turn. Saved values can be stale by up to 15 turns.

```
city[slot].foodBox = 0
city[slot].shieldBox = 0
city[slot].tradeRevenue = 0
city[slot].name = ""

// Assign best initial production item
bestScore = 0; bestItem = 0
for unitType = 2..61:
    if unit_data[unitType].role == LAND_MILITARY(1):
        if can_build(civId, slot, unitType):
            score = (unit.attack * 8) / unit.cost
            if unit.flags & 0x04: score += 1  // veteran bonus
            if score >= bestScore:
                bestScore = score
                bestItem = unitType
civData[civId].unitTypeBuiltCount[bestItem]++

// Assign city name
assign_city_name(slot)

// Clear buildings bitmask
city[slot].buildings = 0x00000000

// First city special handling
if civData[civId].cityCount == 1:
    set_building(slot, PALACE=1, 1)  // auto-build palace
    civData[civId].capitalX = x
    // AI initial exploration
    if not human(civId):
        // Reveal city radius (20 tiles) to owner
        for each of 20 radius tiles:
            reveal_tile(tile, 1 << civId, 1, 1)
            if turn > 39 and tile < 8:
                // Road native terrain if possible
                if tile_owner_bits & humanBitmask == 0 and
                   terrain.foodYield == -2:
                    set_tile_improvement(tile, ROAD=4, 1, 1)
        // Late-game AI start: bigger initial city
        if turn > 40 and no humans around:
            city.size = clamp((turn - 20) / 20, 2, 10)
            set_building(slot, GRANARY=4, 1)
            set_building(slot, MARKETPLACE=5, 1)
            set_building(slot, AQUEDUCT=6, 1)
        // Auto-start production if late game
        if turn > 40 and city.currentProduction != 0:
            start_unit_production(currentProduction, civId, x, y)

// Clear trade routes
city[slot].tradeRouteCount = 0
city[slot].tradePartners = {0, 0, 0}
city[slot].tradeCommodities = {0, 0, 0}

// Check adjacent tiles for special features
for each of 9 adjacent tiles:
    if valid_tile:
        if has_river(tile):
            if terrain_food_yield > 1 and (continent < 63 or city.attribs & 0x80):
                city.attribs |= 0x200000  // river adjacent
            city.attribs |= 0x80  // has water access
        if tile has resource_flag or terrain == mountain(5):
            city.attribs |= 0x800  // has strategic resource

// Update tile ownership + trade desirability
city_update_tile_ownership(slot)
calc_city_trade_desirability(slot)

// Clear derived fields
city[slot].tradeCommodities = {0, 0, 0}
city[slot].foodProduction = 0
city[slot].shieldProduction = 0
city[slot].happyCitizens = 0
city[slot].unhappyCitizens = 0

// Multiplayer: broadcast city creation
if multiplayer > 2:
    send_broadcast(0xFF, CREATE_CITY_NOTIFY, ...)
    flush_send_buffer(5000)

DAT_006ad8c4 = 0
return slot
```
