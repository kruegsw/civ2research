### FUN_00440453 / establish_trade_route
**Category**: GAME_LOGIC.trade | **Size**: 765b | **Ported**: PA

```
establish_trade_route(city1, city2, commodity):
  // Check if route already exists (same partner + same commodity)
  for i = 0 to city[city1].numTradeRoutes - 1:
    if tradePartner[i] == city2 AND tradeRouteType[i] == commodity:
      return  // duplicate, skip

  if city[city1].numTradeRoutes < 3:
    // Open slot available — add directly
    city[city1].numTradeRoutes++
    set_trade_route(city1, newSlot, city2, commodity)
    recalc_city(city1)  // FUN_0043d400
    recalc_city(city2)
  else:
    // All 3 slots full — find lowest-value route to replace
    minValue = 9999
    for i = 0 to numTradeRoutes - 1:
      // Skip if both existing route and new route are negative (actively traded)
      if tradeRouteType[i] < 0 AND commodity < 0:
        return

      partner = tradePartner[i]
      baseTradeValue = city[partner].baseTrade  // at +0x35E
      distance = calc_iso_distance(city1, partner)  // FUN_00488a45

      // Airport bonus: if both cities have building 0x20 (airport) and distance < 2
      if has_building(city1, 0x20) AND has_building(partner, 0x20) AND distance < 2:
        distance = 1

      // Distance multiplier
      if distance > 0:
        baseTradeValue += (distance * baseTradeValue) >> 1

      // Same-owner penalty: halve value
      if city1.owner == partner.owner:
        baseTradeValue >>= 1

      if baseTradeValue < minValue:
        minValue = baseTradeValue
        replaceSlot = i

    // Replace if new route is worth more
    if minValue <= city[city2].baseTrade:
      set_trade_route(city1, replaceSlot, city2, commodity)
      recalc_city(city1)
      recalc_city(city2)
```

**Key refs**: FUN_00488a45 (iso distance), FUN_0043d20a (has_building), FUN_0043d400 (recalc_city)
**Porting notes**: JS reducer.js has ESTABLISH_TRADE action with simplified logic. Airport bonus and replacement logic not fully ported.

  // Difficulty modifiers (raging hordes / flattening)
  if gameFlags & 4:   revenue = (revenue * 4) / 5
  if gameFlags & 8:   revenue = (revenue * 5 + round) / 4

  // Revenue formula: (homeTrade + destTrade) * (distance + 10) / 24
  revenue = (home.baseTrade + dest.baseTrade) * (revenue + 10) / 24

  // Intercontinental bonus: double if different continents
  homeContinent = get_continent(home.x, home.y)
  destContinent = get_continent(dest.x, dest.y)
  if homeContinent != destContinent: revenue *= 2

  if unitOwner == destCityOwner: revenue /= 2

  // Food caravan bonus (type '1' = food): +50%
  if unit.isFoodCaravan: revenue += revenue / 2

  // Route modifier from airports (+0x20) and harbors (+0x19)
  routeModifier = calc_route_bonus(unitOwner, dest, home)
  if has_building(dest, 0x20) AND has_building(home, 0x20):
    if same continent: routeModifier += 1
    else: routeModifier += 2
  if has_building(dest, 0x19): routeModifier += 1
  if has_building(home, 0x19): routeModifier += 1
  revenue += (routeModifier * revenue) / 2

  // Supply/demand bonus (commodity matching)
  supplyDemandBonus = 0
  switch commodityType:
    case 3,5,8,10:     supplyDemandBonus = revenue / 2
    case 9,11,12,13:    supplyDemandBonus = revenue
    case 14:            supplyDemandBonus = (revenue * 3) / 2

      else:
        revenue = (revenue + supplyDemandBonus) * 2

  // Era bonus: pre-industrial doubles
  if turn < 200 AND !hasTech(unitOwner, 0x26) AND !hasTech(unitOwner, 0x39):
    revenue *= 2

  // Tech discounts: Economics (-1/3), Trade (-1/3)
  if hasTech(unitOwner, 0x43): revenue -= revenue / 3
  if hasTech(unitOwner, 0x1E): revenue -= revenue / 3

  // Cap revenue with research randomizer
  capValue = 30000
  randomVal = rand() % 10 + 200
  researchCap = calcResearchCost(unitOwner, randomVal, capValue)
  revenue = clamp(revenue, 0, (researchCap * 2) / 3)

  // Show popup messages (CARAVAN / FOODCARAVAN / CARAVANOTHER)
  // ... (extensive UI messaging code, MP-aware)

  // Update diplomatic attitude
  update_attitude(destOwner, unitOwner, -10)  // FUN_00456f20
  update_attitude(unitOwner, destOwner, -10)

  // Apply revenue
  if commodityType < 0:  // Food caravan
    recalc_city(dest)
    dest.foodInBox += ((dest.size + 1) * foodToGrowBase) / 2
  else:  // Trade caravan
    civ[unitOwner].gold += revenue
    civ[unitOwner].researchAccum += revenue

  // Disband unit
  disband_unit(unitIdx, 1)  // FUN_005b6042

  // Establish trade route between home and dest
  establish_trade_route(homeCityId, destCityIdx, commodityType)

  // Establish reverse route: pick best available commodity from dest
  if commodityType >= 0:
    bestScore = 0
    bestCommodity = -1
    for i = 0 to 2:

      if commodity != commodityType:

    // ... (messaging)
```

change_city_production(cityIdx, newProduction):
  oldProduction = city[cityIdx].currentProduction  // byte at +0x379
  civId = city[cityIdx].owner

  // Decrement old building count if it was a valid building type
  if oldProduction >= 0 AND oldProduction < 62:
    civ[civId].buildingCounts[oldProduction] -= 1

  // Auto-select production if param > 98
  if newProduction > 98:
    newProduction = AI_choose_production(cityIdx, 0, 0)  // FUN_00498e8b

  // Set new production (if < 99)
  if newProduction < 99:
    if humanPlayer AND !cheatMode:
      estimate shields remaining
    city[cityIdx].currentProduction = newProduction

  // Handle wonder-related events (production values < -38 are wonders)
  // Production index: wonderIdx = abs(production) - 39
  if oldProduction != newProduction AND !humanPlayer:
    if newProduction < -38 OR oldProduction < -38:
      // Count other cities building same wonder
      sameOldCount = 0
      sameNewCount = 0
      for each city of same owner:
        if building oldProduction: sameOldCount++
        if building newProduction: sameNewCount++

    if newProduction < -38:  // Starting a wonder
      wonderIdx = abs(newProduction) - 39

      if oldProduction < -38:
        // SWITCHING between wonders
        if sameNewCount == 0:
          // Transfer shields from old wonder
          city.shieldsInBox /= 2
          if sameOldCount == 0 AND wonderCityIds[oldWonderIdx] == -1:
            // Reassign other cities building old wonder to new
            for each city building oldWonder: change to newProduction
          show "SWITCHWONDER" dialog
          send MP packet 0x13

      else:
        // STARTING a new wonder (from non-wonder)
        if !civ.wonderStartedFlag[wonderIdx]:
          civ.wonderStartedFlag[wonderIdx] |= 1
          show "STARTWONDER" dialog
          send MP packet 0x12

      // Free shields bonus for first-in-era wonder completion
      if city.shieldsInBox == 0:
        eraGroup = (wonderIdx - 39) / 7
        if !civ.eraWonderBonus[eraGroup]:
          allCivsHaveBonus = true
          for each alive civ: check eraWonderBonus[eraGroup]
          if allCivsHaveBonus:
            freeShields = (difficulty + eraWonderBonus * -2)
            if freeShields < 1: freeShields = 0
            city.shieldsInBox += freeShields * wonderCost

    else if oldProduction < -38 AND sameOldCount == 0:
      // ABANDONING a wonder
      oldWonderIdx = abs(oldProduction) - 39
      if wonderCityIds[oldWonderIdx] != cityIdx AND !wonderComplete AND !abandonedFlag:
        civ.wonderFlags[oldWonderIdx] |= 2
        show "ABANDONWONDER" dialog
        send MP packet 0x14

  // Increment new building count
  if newProduction >= 0 AND newProduction < 62:
    civ[civId].buildingCounts[newProduction] += 1
```
