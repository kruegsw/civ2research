### FUN_0059062c — move_unit
- **Category**: GAME_LOGIC.Movement
- **Size**: 17963 bytes (lines 49-1280)
- **Ported**: PA (probabilistic movement check only — reducer.js line 649)
- **Signature**: `uint move_unit(int unit_idx, uint direction, byte flags)`

**Pseudocode**:
```
function move_unit(unit_idx, direction, flags):
    // --- GUARD ---
    if reentrancy_guard != 0 OR unit dead OR coords < 0:
        return 0
    reentrancy_guard = 1
    save flags to DAT_006ad0cc

    LOOP:  // outer loop handles landfall re-entry
        clear move state variables
        src_x, src_y = unit position
        if direction >= 0:
            dst_x = wrap(src_x + dx[direction])
            dst_y = src_y + dy[direction]

        domain = unit_type[unit.type].domain   // 0=land, 1=sea, 2=air
        owner = unit.owner
        src_terrain = get_terrain(src_x, src_y)
        dst_terrain = get_terrain(dst_x, dst_y)
        is_ocean_src = (src_terrain == 10)
        is_ocean_dst = (dst_terrain == 10)
        dst_owner = get_tile_effective_owner(dst_x, dst_y)
        dst_unit = get_top_unit(dst_x, dst_y)
        dst_city = get_city_at(dst_x, dst_y)

        // --- MULTIPLAYER MAP LOCK ---
        if save_version > 2 AND mp_enabled AND human_civ AND direction >= 0:
            if mp_lock_map(owner, src, dst) fails:
                goto CLEANUP

        // === LAND UNIT SPECIAL CASES (domain == 0) ===

        // 1. ZOC check: if no unit at dst, check ZOC blocking for both src & dst
        if domain == 0 AND dst_unit < 0:
            if blocked by ZOC on both sides AND not on ocean AND no amphibious flag:
                show "cannot move" message; goto FAIL

        // 2. Diplomat/Spy entering enemy city (role == 7)
        if unit_role == 7 AND dst_city >= 0:
            if dst_city.owner != owner AND dst_city.owner is human:
                initiate diplomat_vs_city(unit, city)
                goto CLEANUP

        // 3. Caravan/Freight entering city (role == 6)
        if unit_role == 6 AND dst_owner >= 0 AND dst_owner != owner:
            if dst_city >= 0:
                caravan_trade(unit, city); goto CLEANUP
            else:
                caravan_vs_unit(unit, dst_unit); goto CLEANUP

        // 4. Allied repair: friendly city with alliance treaty, unit damaged
        if dst_city >= 0 AND dst_city.owner != owner AND treaty[owner][dst_owner] has alliance:
            deduct HP proportional to repair cost
            show "ALLIEDREPAIR"; goto CLEANUP

        // 5. Land unit vs ocean tile with no city
        if domain == 0 AND is_ocean_dst:
            if dst_owner == owner AND can_load_transport(dst_unit, role=6):
                set bVar5 (transport boarding); continue
            else: goto FAIL

        // 6. Enemy unit on tile: diplomat expel logic
        if dst_unit >= 0 AND dst_unit.owner != owner:
            // Diplomat expel (role 6): find best city to expel to, show dialog
            // "EXPEL" / "UPMINE" / "UPYOURS" / "UPYOURSTOO" messages
            // May trigger war declaration via spaceship_ai_should_start check
            // Sets treaty hatred/war flags
            goto CLEANUP

        // === MOVEMENT COST CALCULATION ===
        remaining_moves = get_remaining_moves(unit)
        if remaining_moves < COSMIC[0]:  // not enough moves
            if human AND interactive:
                show "FATIGUE" dialog — user can cancel
            else if AI:
                if remaining_moves < COSMIC[0]-1: goto FAIL

        // Calculate terrain movement cost:
        if domain == 0 (land):
            if both src & dst have road/railroad:
                cost = railroad ? 0 : road_cost (1 third)
            else if unit has all-terrain flag:
                cost = 1
            else if both have river AND adjacent:
                cost = 1
            else:
                cost = terrain_table[dst_terrain] * COSMIC[0]
        else:  // sea or air
            cost = COSMIC[0]

        // === PROBABILISTIC MOVEMENT CHECK ===
        // (This is the part ported in reducer.js line 649)
        if remaining_moves < cost AND domain == 0 AND unit.moved_this_turn:
            if human OR interactive:
                random = rand() % cost
                if remaining_moves <= random:
                    exhaust unit moves; mark as moved
                    goto CLEANUP

        // === FLAG-BASED BRANCHING ===
        if flags & 2 == 0:  // normal move
            result = attempt_combat_or_move(unit, direction, 1)
            // attempt_combat_or_move calls FUN_00580341
            set local_f0 based on result
        else:  // AI evaluation mode (flags & 2)
            result = attempt_combat_or_move(unit, direction, 1)
            if success: set local_4c = 1, loop back

        // === COMBAT VS CITY (sea/air units attacking city) ===
        if dst_city >= 0 AND dst_city.owner != owner:
            if domain != 0 (sea/air) AND attack < threshold:
                show "OCCUPY" message (can't capture)
            else if flags & 2:
                call FUN_0057f9e3 for combat evaluation
            else:
                mark as moved; goto CLEANUP

        // === ACTUAL MOVEMENT EXECUTION ===
        // Check visibility for each civ (8 players), build arrays:
        //   aiStack_a0[civ] = 1 if this civ can see the move
        //   aiStack_80[civ] = 1 if this civ should see combat animation

        // Active player special: check Shift key for skipping animation
        // GetAsyncKeyState(0x10) — shift key suppresses animation

        if not multiplayer_realtime OR hotseat:
            // Local execution path:
            unload_from_transport(unit)
            broadcast move notifications to visible civs
            if active player can see: play sound, animate
            update unit stack, set new position
            call FUN_005b3ae0(unit, dst_x, dst_y, 0)
        else:
            // Networked MP path:
            set DAT_006c914c = -2 (battle sync state)
            send move message (0x5A) to server
            broadcast to visible civs
            wait up to ~3600 ticks for server response (DAT_006c914c == 1)
            if timeout: show "SERVERCONNECTTIME" error

        // === POST-MOVE PROCESSING ===
        // Update continent tracking for civ
        // Combat resolution if enemy at destination
        if dst_city >= 0 AND dst_city.owner != owner:
            unit.moves_remaining = 0
            call FUN_0057b5df(city, owner, 0)  // initiate city combat

        // Barbarian visibility update
        // Wake sentry units in stack if landing on ocean

        // Air unit landing/fuel check:
        if domain == 1 (sea/air with fuel):
            check for city or carrier at destination
            if no valid landing site: increment fuel counter
            if fuel exhausted: destroy unit ("crashed at sea" etc.)

        // Trireme sinking check:
        if unit has trireme flag AND out of moves AND no lighthouse:
            1/COSMIC[1] chance of sinking per turn
            boosted by Navigation tech (2x) and Magnetism tech (2x)
            if sinking: destroy unit, show "TRIREME" message

        // Goody hut processing:
        if goody_hut_at_dst AND domain == 0:
            call FUN_0058f040(unit)  // process goody hut

        // Air unit auto-wake on arrival in non-friendly territory

        // Transport boarding from shore (bVar5 path):
        if transport_boarding:
            unload from current stack
            find matching transport in dst stack
            board transport

        // LONGMOVE tracking (human player):
        if unit has goto order: increment move counter
            if counter > 47: show "LONGMOVE" dialog (cancel option)

        // Update last_direction on unit record

    CLEANUP:
        if mp_locked: mp_unlock_map(owner)
        if save_version > 2: flush send buffer
        if turn changed during move: reentrancy_guard = 0; return 0

        // On failure: clear orders, set to idle
        // On success: clear "was fortified" flag
        reentrancy_guard = 0
        return local_f0  // 1 = moved, 0 = failed
```

### FUN_00594d42 — mp_lock_map
- **Category**: GAME_LOGIC.Movement (Multiplayer)
- **Size**: 971 bytes (lines 1289-1378)
- **Ported**: NN (Civ2-specific XDaemon MP locking; JS uses server-authoritative model)
- **Signature**: `u4 mp_lock_map(int civ, int from_x, int from_y, int to_x, int to_y, int wait)`

**Pseudocode**:
```
function mp_lock_map(civ, from_x, from_y, to_x, to_y, wait):
    set lock_active flag
    clear lock record for civ

    if hotseat_mode:
        // Check all 7 other civs' lock records for collision
        for each other_civ:
            if other_civ's locked (from OR to) coords overlap our (from OR to):
                notify other player, return 0 (locked by other)
        // No collision: store our from/to in lock record
        store lock coords; notify if needed; return 1
    else:
        // Networked: send lock request msg 0x51 via XDaemon
        send_message(0x51, 0, civ, from_x, from_y, to_x, to_y)
        flush_send_buffer(60000)
        if wait == 0: return 1 (fire-and-forget)
        // Wait up to ~57 seconds for server response
        wait_loop(~0xe0f ticks, check lock_record[civ] != 0)
        if timeout: log "LockMap: Connection to server timed out"; return 0
        return lock_record[civ] == 1 ? 1 : 0
```

**Key References**:
- Globals: DAT_006ad8cc (lock_active), DAT_0064ba48 (lock records, stride 0x18, 8 civs), DAT_006ad2f7 (is_hotseat)

```
function spaceship_get_max_component(civ, type):
    max = spaceship_max_array[type * 3]   // DAT_00634f64, stride 0x0C

    if type == 1 OR type == 2:  // propulsion or fuel
        n = (civ_spaceship_structurals[civ] + 1) - (type * 2 - 2)
        max = clamp(max, 0, (n + round_toward_zero) / 4)

        max = clamp(max, 0, (n + round_toward_zero) / 8)

    return max
```

```
function spaceship_recalc_stats(civ, apply_penalties):
    // Check Apollo Program tech (0x20) — sets "has apollo" flag on civ
    if not launched:
        if civ has_tech(0x20): civ_ss[civ].flags |= 8  // Apollo bonus

    flags = civ_ss[civ].flags

    // 1. MASS: sum(component_count[i] * component_weight[i]) for i=0..5
    mass = 0
    for i = 0..5:
        mass += civ_ss[civ].components[i] * ss_weight_table[i]
    // Also compute "mass if one more of each empty slot"
    mass_with_extras = mass
    for i = 0..5:
        if civ_ss[civ].components[i] == 0: mass_with_extras += ss_weight_table[i]
    DAT_006ad0e4 = mass

    // 2. SUCCESS PROBABILITY (starts at 100%)
    success = 100

    // 3. FUEL RATIO: clamped_life / clamp(clamped_hab, 1, 99) * 100
    fuel_ratio = (get_clamped(civ, 4) * 100) / clamp(get_clamped(civ, 3), 1, 99)
    DAT_006ad0f0 = fuel_ratio
    if apply_penalties:
        success = (clamp(fuel_ratio, 0, 100) * success) / 100

    // 4. ENERGY RATIO: clamped_solar * 200 / clamp(hab + life, 1, 99)
    energy_ratio = (get_clamped(civ, 5) * 200) / clamp(get_clamped(civ, 3) + get_clamped(civ, 4))
    DAT_006ad0e8 = energy_ratio
    success = (clamp(energy_ratio, 0, 100) * success) / 100

    // 5. POPULATION RATIO
    pop_ratio = (get_clamped(civ, 2) * 100) / clamp(get_clamped(civ, 1), 1, 99)
    DAT_006ad0dc = pop_ratio

    // 6. FLIGHT TIME: mass * COSMIC[20] / (pop_capacity * 10 + 1)
    flight_constant = COSMIC[20]  // DAT_0064bcdc
    if has_apollo: flight_constant = flight_constant * 3 / 4
    // Scale down large constants
    scale = 1; while flight_constant > 100: flight_constant >>= 1; scale <<= 1

    pop_cap = spaceship_calc_population_capacity(get_clamped(civ, 2))
    thrust = spaceship_calc_population_capacity(get_clamped(civ, 1), 0, pop_cap * 10)
    flight_time = (mass * flight_constant) / (clamp(thrust * 10) + 1)

    // Also compute worst-case and raw-components flight times
    // Apply scale factor back
    if scale > 1: flight_time *= scale

    DAT_006ad0f4 = flight_time

    // 7. Penalty for long flight: if flight_time > 150 (15.0 turns), -1% per turn over
    if flight_time > 0x96:
        success -= (flight_time - 0x96) / 10
    success = clamp(success, 0, 100)
    DAT_006ad0ec = success

    // 8. Write arrival turn estimate
    if not launched:
        civ_ss[civ].hab_count = get_clamped(civ, 3)
        civ_ss[civ].launch_year = current_year_index
        civ_ss[civ].arrival_turn = current_year_index + flight_time / 10
        DAT_006ad0e0 = current_year_index + worst_case_flight_time / 10
```
