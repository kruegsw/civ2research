fn mp_join_game(mode):
    // mode: 0=IP connect, 1=browse sessions, 2=reconnect host, 3=reconnect client
    setup_seh_frame()
    init_mp_subsystems()

    if mode == 2 || mode == 3:
        // RECONNECT PATH
        result = XD_OpenConnection(connStr, timeout_secs * 1000)
        if result < 0:
            if connType != IPX && connType != serial:
                show_error("Connection failed")
            cleanup_and_return()
    else if mode == 0:
        // DIRECT IP PATH
        if lobbySessionId == -1:
            // Show IP entry dialog "IPOFGAME"
            show_dialog(DIALOG_CONTEXT, "IPOFGAME", 0x1F)
            result = run_dialog()
            if result == -1: cleanup_and_return()
            if strlen(ipString) == 0: goto browse_path
            result = XD_OpenConnection(connStr, timeout_secs * 1000)
            if result < 0: show_error(); cleanup_and_return()
            // If lobbySessionId != -1, send lobby messages
            if lobbySessionId != -1:
                XD_LobbySendMessage(1)  // join
                XD_LobbySendMessage(3)  // confirm
            // Wait for server with "WAITINGFORSERVER" dialog
            show_wait_dialog("WAITINGFORSERVER")
            if result == -1: cleanup_and_return()
            if gameState != -1:
                // Server responded — verify version
                verify_version()
                goto post_connect
            // Retry up to 3 times
            if lobbySessionId == -1 || retries > 3:
                show_error("Connection lost")
                cleanup_and_return()
            XD_CloseConnection()
        // fall through to browse
    else:
        // mode == 1: BROWSE PATH
        create_session_browser_ui()
        loop:
            enable_session_refresh_timer()
            userChoice = run_dialog()
            disable_timer()
            if userChoice < 0: cleanup_and_return()
            // Find selected session in linked list
            session = find_session_by_id(userChoice)
            if session == NULL: continue

            occupiedSlots = popcount(session.playerBitmask_B4)
            totalSlots = popcount(session.playerBitmask_B5)
            if totalSlots >= maxPlayers || totalSlots >= occupiedSlots || session.version == 0:
                show_error("Game full")
                continue
            // VERSION CHECK: compare session.versionStr with "5_4_0f Multiplayer 26 March 99"
            if session.versionStr != EXPECTED_VERSION:
                show_error("Version mismatch")
                continue
            break  // version OK, session joinable

    // POST-CONNECT: Exchange game profile
    post_connect:
    save_tick_count()
    if mode != 0: sync_mp_state()
    show_wait_dialog("WAITINGFORSERVER")
    if result == -1: send_disconnect(); cleanup_and_return()
    check_connection_errors()
    if result == -2: show_error("Timeout"); cleanup_and_return()

    sync_game_data()
    // Wait to join: "WAITTOJOIN"
    show_wait_dialog("WAITTOJOIN")
    if result != -1 && no_errors:
        if connType > 0 && scenarioPath != "":
            chdir(scenarioPath)
            reload_rules_for_scenario()
            if connType == IPX || (scenarioFlags & 0x80):

                for civ 1..7:

                    set_civ_graphics(civ)

                reload_art_and_music()
            // Copy game state from session info
            scenarioFlags_local = gameFlags & 0xFFFF7FFF
            scenarioYear = gameYear
            // Exchange game profile dialog
            show_game_profile_dialog()

        result = mp_setup_game_profile(isJoining=1)
        if result == -1: disconnect(); cleanup_and_return()

        // Load save or start new game
        if noSaveFile && noScenario && gamePhase < 2:
            result = start_new_game()
        else:
            result = load_save_file()
        if result != 0: disconnect(); cleanup_and_return()

        send_message(JOIN_CONFIRM)
        sync_game_data()
        show_wait_dialog("WAITINGONJOIN")
