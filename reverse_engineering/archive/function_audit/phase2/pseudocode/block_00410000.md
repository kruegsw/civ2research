            send_mp_message(0x33, 0xff, viewport_x, viewport_y, current_turn, ...)

    if mp_mode > 2:
        // Internet multiplayer setup
        init_timer(0x120, 1)
        set_mp_callback(LAB_00403c74)
        if is_host():
            send_sync(0xff)
        mp_active_flag = 1
        XD_FlushSendBuffer(180000)
        start_mp_thread()

    post_load_game_init()

    // Show simultaneous tutorial if first time
    if simultaneous_mode and mp_mode > 2 and mp_current_turn != 0 and tutorial_shown == 0:
        show_popup("SIMULTUT", msg_buf, 0)
        tutorial_shown = 1

    record_timestamp(timer_buf)
    game_started_flag = 1

    // Enter main game loop based on mode
    if mp_mode < 3:
        run_hotseat_game_loop()
    else:
        if not is_returning_player:
            XD_FlushSendBuffer(5000)
            // Wait for all players ready
            while not all_players_ready:
                process_mp_messages(1, 0)
        else:
            // Returning player: send ready signal
            send_mp_message(0x9c, 0xff, 0, 0, 0, ...)
            XD_FlushSendBuffer(5000)

        // Run appropriate game loop
        if not simultaneous_mode:
            if not is_returning_player:
                result = run_sequential_mp_loop()
                if result != 0:
                    run_sequential_mp_endgame(1)
            else:
                run_sequential_mp_endgame(0)
        else:
            if not is_returning_player:
                result = run_simultaneous_mp_loop()
                if result != 0:
                    run_simultaneous_mp_endgame()
            else:
                run_simultaneous_mp_endgame()

    save_preferences()
    clear_mp_callback()
    reset_wonder_data()
    cleanup_network(1)
    cleanup_network_2()

    if mp_mode > 2:
        // Internet cleanup
        close_network()
        cleanup_string_object()  // via FUN_004201ef
        destroy_mp_resources()
        if is_returning_player:
            reset_mp_rankings(1)
        // Write chat log
        file = open("chatlog.txt")
        if file:
            write(chat_buffer, strlen(chat_buffer))
            close(file)
        // Clear ranking data
        for civ in 0..7:
            clear_rankings(civ)
        clear_global_rankings()
        init_mp_rankings()
        cleanup_mp_rankings()
        // Destroy extra string object

    cleanup_advisors()
    cleanup_game_views()  // twice for safety
    cleanup_game_views()

    if shutdown_requested:
        // SEH cleanup and return
        return

    cleanup_game_views()
    if lobby_launch_mode != -1:
        // Lobby-launched: exit after one game
        return

// End of main loop
shutdown_subsystems()
