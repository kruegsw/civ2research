## FUN_005b2a39 / calc_unit_movement_points

`uint calc_unit_movement_points(int unit_id) → movement_points`

### Pseudocode
```

base_moves = unit_types[type_id].move_rate   // utype+0x0A
if base_moves == 0: return 0

owner = units[unit_id].owner
domain = unit_types[type_id].domain          // utype+0x01

if domain == 2:  // AIR
  // Nuclear Power tech (0x3B): +COSMIC
  if civ_has_tech(owner, 0x3B):
    base_moves += COSMIC                     // DAT_0064bcc8
  // Magellan's Expedition wonder (0xC): +2*COSMIC
  if civ_has_wonder(owner, 0xC):
    base_moves += COSMIC * 2
  // Lighthouse wonder (3): +COSMIC if no flag 0x20
  if civ_has_wonder(owner, 3):
    if not (unit_types[type_id].flags1 & 0x20):
      base_moves += COSMIC

result = base_moves

// Damage-based reduction (if "hitpoint recovery" flag set)
if units[unit_id].damage != 0 AND (DAT_00655ae8 & 0x10) != 0:
  if domain != 1:  // not SEA (sea units unaffected)
    max_hp = get_unit_max_hp(unit_id)
    if max_hp < 2: max_hp = 1
    hp_remaining = get_unit_hp_remaining(unit_id)
    reduced = (hp_remaining * base_moves) / max_hp
    // Round up to next COSMIC multiple
    if reduced % COSMIC != 0:
      reduced += COSMIC - (reduced % COSMIC)
    // Minimum: COSMIC for land, 2*COSMIC for air
    minimum = COSMIC * (2 if domain == 2 else 1)
    result = max(reduced, minimum)
```
