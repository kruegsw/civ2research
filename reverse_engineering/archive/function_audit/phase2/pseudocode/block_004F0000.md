function calc_building_upkeep_cost(civ_id, building_id):
    cost = improvement_table[building_id].upkeep_cost   // DAT_0064c48d[id*8]

    if building_id == 2:  // Courthouse
        if difficulty < 2 AND cost > 0:
            cost -= 1
        // Walk tech chain: find last tech in chain starting from tech 0x35
        last_tech = 0x35
        while last_tech >= 0 AND tech_table[last_tech].name == '':
            last_tech = tech_table[last_tech].prereq    // chain pointer at +0xE
        if civ_has_tech(civ_id, last_tech):

            cost += 1

    if cost == 1:
        if civ_has_active_wonder(civ_id, 0x11):  // United Nations
            cost = 0

        if building_id == 4 OR building_id == 0x0E OR building_id == 0x0B:
            cost = 0   // Temple, University, Library free under Fundamentalism

    return cost

function handle_city_expansion(city_idx):

    // First: check city radius for pollution tiles
    for i = 0 to 20:  // 21 tiles in city radius
        tx = city[city_idx].x + radius_dx[i]
        ty = city[city_idx].y + radius_dy[i]

        if expansion_site_found == 0:

            if settler_priority > 2:

                    settler_priority += 1
                if city_has_building(city_idx, BARRACKS):
                    settler_priority += 1
            if city[city_idx].size < 4:

            city[city_idx].flags |= 0x80000

- **Address**: 004FA617 | **Size**: 240B
- **Classification**: GL.EVENT | **Porting**: TH

```
function event_alloc_node():
    // Find tail of doubly-linked list
    tail = this.event_list_head  // +0x30C
    if tail != NULL:
        while tail.next != NULL:   // +0x1BC = next pointer
            tail = tail.next

    // Allocate from memory pool
    node = pool_alloc(this.pool, 0x1C4)  // 452-byte node
    if node == NULL:
        return NULL

    memset(node, 0, 0x1C4)
    if tail == NULL:
        this.event_list_head = node
        node.next = NULL
        node.prev = NULL
    else:
        tail.next = node
        node.prev = tail
        node.next = NULL

    this.event_count++  // +0x308
    return node
```
- **Key refs**: `thunk_FUN_00498159` (pool_alloc), event node layout: +0x1BC=next, +0x1C0=prev

function event_dispatch_actions(event_node):
    // Guard: JUSTONCE + already fired
    if (event_node.flags[0] & 0x40) != 0 AND (event_node.flags[1] & 0x20) != 0:
        return 0  // already fired, skip

    // Dispatch actions based on bitflags in flags[0] and flags[1]

    if flags[0] & 0x80:  play_cd(event_node)            // PLAYCDTRACK

    if flags[1] & 0x08:  give_tech(event_node)           // GIVETECHNOLOGY
