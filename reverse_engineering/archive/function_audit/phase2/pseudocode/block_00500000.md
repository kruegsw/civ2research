```
function citywin_click_resource_map(screen_x, screen_y):
    cityIdx = read_DAT_006a91b8()
    if city.owner != activePlayer && !godMode: return
    calc_city_production(cityIdx, 0)
    tile_w = scale(0x40, sprite_zoom)
    tile_h = scale(0x20, sprite_zoom)
    // Convert screen coords to resource map-relative coords
    rel_x = screen_x - map_origin_x
    rel_y = screen_y - map_origin_y - tile_h/2
    if out of bounds: return
    // Coarse grid position
    grid_dx = (rel_x / tile_w) * 2 - 3
    grid_dy = (rel_y / tile_h) * 2 - 3
    // Fine sub-tile adjustment via diamond hit-test (FUN_005c0bf2)
    quadrant = diamond_hittest(rel_x % tile_w, rel_y % tile_h)
    if quadrant != 0:
        grid_dx += CitySpiralAdjDX[quadrant]
        grid_dy += CitySpiralAdjDY[quadrant]
    // Find matching spiral slot (0..20)
    slot = -1
    for i in 0..20:
        if CitySpiralDX[i] == grid_dx && CitySpiralDY[i] == grid_dy:
            slot = i; break
    if slot < 0: return
    if slot == 0x14:  // slot 20 = center tile reset
        city.workers_field = 0
    else:
        is_working = check_tile_worked(cityIdx, slot)
        if is_working:
            set_tile_worked(cityIdx, slot, 0)  // remove worker
            auto_reassign_workers(cityIdx, 1)
        else:
            if tile_blocked[slot]: return  // another city working it
            if specialist_count == 0:
                city.workers_field = 0  // reset
            else:
                set_tile_worked(cityIdx, slot, 1)  // add worker
                auto_reassign_workers(cityIdx, -1)
    refresh_top_panels(0)
```

```
        cost = unit_type_table[unit_type].cost
        remaining = max(0, cost * COSMIC[4] - city.shield_box)
        buy_cost = (remaining^2 / 20) + remaining * 2
    if city.shield_box == 0:
        buy_cost *= 2  // double if nothing accumulated

    can_afford = (buy_cost <= civ.treasury)

    if can_afford && choice == 0 && !close_requested:
        civ.treasury -= buy_cost
        city.shield_box = cost * COSMIC[4]  // fully funded
        recalc_city(cityIdx)
        play_sound(0x68, 1)
        calc_city_production(cityIdx, 1)
```
