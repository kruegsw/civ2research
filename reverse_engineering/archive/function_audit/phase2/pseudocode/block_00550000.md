## Function 68: FUN_0055a980 — `build_trade_route_map`

### Pseudocode

```
function build_trade_route_map():
    DAT_0062d040 = 1                            // computing flag

    for pass = 0 to 1:                          // 0=land, 1=sea
        bitmap = (pass == 0) ? trade_bitmap_land : trade_bitmap_sea
        memset(bitmap, 0, minimap_w * minimap_h)

        for mx = 0 to minimap_w - 1:           // DAT_006d116a
            for my = 0 to minimap_h - 1:        // DAT_006d116c
                // Convert minimap coords to map coords
                tileX = mx * 4 + 1
                tileY = my * 4 + 1

                // Find valid trade tile at this minimap cell
                bodyId = find_trade_tile(tileX, tileY, &outX, &outY, pass)
                if bodyId < 0: continue

                // Check 4 diagonal neighbors (NE, SE, SW, NW)
                for dir = 0 to 3:
                    nx = wrap_x(tileX + diagonal_dx[dir] * 4)
                    ny = tileY + diagonal_dy[dir] * 4

                    if not is_tile_valid(nx, ny): continue

                    neighborBody = find_trade_tile(nx, ny, &nOutX, &nOutY, pass)
                    if neighborBody != bodyId: continue

                    // Check if trade path exists (max cost 5)
                    pathCost = pathfind_trade(outX, outY, nOutX, nOutY, pass, 5)
                    if pathCost <= 0 OR pathCost >= 5: continue

                    // Set direction bit in this cell
                    bitmap[my * minimap_w + mx] |= (1 << dir)

                    // Set reverse direction bit in neighbor cell
                    nmx = wrap_minimap_x(mx + diagonal_dx[dir])
                    nmy = my + diagonal_dy[dir]
                    if nmx >= 0 AND nmy >= 0 AND nmx < minimap_w AND nmy < minimap_h:
                        bitmap[nmy * minimap_w + nmx] |= (1 << ((dir - 4) & 7))

            mapgen_progress_tick()

    DAT_0062d040 = 0                            // done

           AND civ[civB].totalAtkDefSum < civ[thirdCiv].totalAtkDefSum

        if (treaty[thirdCiv][civA] & 0x08) AND (treaty[thirdCiv][civB] & 0x08)
           AND is_human(thirdCiv):

            if civ[civA].powerRank < civ[thirdCiv].powerRank

                allyScore += 1                             // very powerful mutual ally

        // If allied with civA but at war with civB — unreliable ally
        if (treaty[thirdCiv][civA] & 0x08)
           AND (treaty_byte1[thirdCiv][civB] & 0x20)
           AND is_human(thirdCiv):
            allyScore -= 1

        // Superpower (rank 7) at war with civB — intimidation
        if civ[thirdCiv].powerRank == 7 AND game_turn > 199
           AND (treaty_byte1[thirdCiv][civB] & 0x20):
            allyScore -= 1

    // Bonus for no vendetta with target
    if not should_declare_war_helper(civA, civB):
        allyScore += 1

    // Bonus if already at war
    if treatyByte0 & 0x08:
        allyScore += 1

    // ── Per-continent strength comparison ──
    ourStrength = 0
    theirStrengthPlusDefense = 1                // avoid div-by-zero
    continentCount = 0

    for continent = 1 to 62:

        if civ[civA].continent_goals[continent] != 0
           AND (civ[civA].continent_atk[continent] + civ[civA].continent_mil[continent]
                < civ[civB].continent_atk[continent]):
            return 0                             // outgunned on a continent we care about

            ourStrength += civ[civA].continent_atk[continent]
            theirStrengthPlusDefense += (civ[civB].continent_atk[continent] >> 1)
                                      + civ[civB].continent_mil[continent]

    patience = personality_table[civ[civA].researchTarget * 0x30]

    else:

    // ── Evaluate exchange ──
    if foundFlags == 3:
        // Both have something to trade — fair exchange
        grant_advance(civA, bestA_tech, civB)
        grant_advance(civB, bestB_tech, civA)
        return 1

    elif foundFlags == 1:
        // Only civA wants something — demand tribute if allied + tech gap
        if (treaty[civA][civB] & 0x08)                         // alliance
           AND (civ[civA].techLevel + (6 - difficulty) * 2 < civ[civB].techLevel
                OR isSuperiorCivActive)
           AND not (treaty_byte2[civA][civB] & 0x04):          // not already demanded
            set_treaty_flag(civA, civB, 0x40000)               // demand tribute
            grant_advance(civA, bestA_tech, civB)              // give tech anyway
            return 1

    elif foundFlags == 2:
        // Only civB wants something — same logic mirrored
        if (treaty[civB][civA] & 0x08)
           AND (civ[civB].techLevel + (6 - difficulty) * 2 < civ[civA].techLevel
                OR isSuperiorCivActive)
           AND not (treaty_byte2[civB][civA] & 0x04):
            set_treaty_flag(civA, civB, 0x40000)
            grant_advance(civB, bestB_tech, civA)
            return 1

    return 0
```

## Function 93: FUN_0055d685 — `check_join_war`

- **Category**: AI.diplomacy

function check_join_war(attacker, joiner, defender) -> bool:
    // ── Blocked conditions ──
    // If joiner already at war with attacker (vendetta 0x2000) or alliance (0x08)
    if treaty[attacker][joiner] & 0x2008:
        return 0

    // ── Check if joiner has vendetta against defender ──
    if not (treaty_byte1[joiner][defender] & 0x20):
        // No vendetta — check if contact exists and both know joiner
        if (treaty[attacker][joiner] & 0x10) AND (treaty[defender][joiner] & 0x10):
            // Set "approached about war" flag for both directions
            treaty[attacker][joiner] |= 0x20
            treaty[defender][joiner] |= 0x20
        return 0

    // ── Joiner is vendetta against defender — evaluate joining ──
    if is_human(joiner):
        // Additional checks for human players
        if not (treaty[attacker][joiner] & 0x10):     // no embassy
            // Check last contact timing
            lastContact = civ[joiner].lastContact[attacker]
            if lastContact - game_turn < 6:
                return 0                                // too recent
            // Weak civs don't always join
            if civ[joiner].powerRank < 7:
                if rand() % 3 != 0:
                    return 0

        // Update last contact turns
        civ[joiner].lastContact[attacker] = game_turn
        civ[joiner].lastContact[defender] = game_turn

    // Show "JOINWAR" message
    set_text_param(0, get_civ_name(attacker))
    set_text_param(1, get_civ_name(defender))
    set_text_param(2, get_civ_name(joiner))
    show_message("JOINWAR")

    // Set vendetta flag
    set_treaty_flag(attacker, joiner, 0x2000)
    return 1
```

## Function 94: FUN_0055d8d8 — `process_diplomatic_contact`

- **Address**: 0x0055D8D8 | **Size**: 7326 bytes
- **Category**: AI.diplomacy

### Pseudocode

```
function process_diplomatic_contact(civA, civB, mapX, mapY):
    // ── Guard clauses ──
    if civA == 0 OR civB == 0: return          // barbarians don't negotiate
    if game_turn == 0: return                  // no diplomacy on turn 0

    // Scenario check
    if scenario_active AND not check_scenario_diplomacy(civA, civB):
        return

    // ── Route based on player types ──
    bothHuman = is_human(civA) AND is_human(civB)

    // ═══ CASE 1: At least one AI civ (SP or MP with AI) ═══
    if not bothHuman:

        // ── MP routing: redirect to remote human player ──
        if multiplayer AND is_human(civA) AND civA != current_player:
            // Check contact freshness: no contact, or expired (>15 turns)
            if (treaty[civA][civB] & 1) == 0
               OR (treaty_byte2[civA][civB] & 1) != 0
               OR lastContact[civA][civB] < 0
               OR game_turn - lastContact[civA][civB] > 15:
                send_mp_event(0x99, civA, civB, mapX, mapY)
                return

        // Same for civB
        if multiplayer AND is_human(civB) AND civB != current_player:
            // (same contact freshness check)
            send_mp_event(0x99, civB, civA, mapX, mapY)
            return

        // ── Local human player diplomacy screen ──
        if is_human(civA) AND civA == current_player:
            if (treaty[civA][civB] & 1) == 0 OR contactExpired:
                if multiplayer AND civA != current_player: return
                open_diplomacy_screen(civA, civB, mapX, mapY, 0)

        if is_human(civB) AND civB == current_player:
            if (treaty[civB][civA] & 1) == 0 OR contactExpired:
                if multiplayer AND civB != current_player: return
                open_diplomacy_screen(civB, civA, mapX, mapY, 0)

    // ═══ CASE 2: Both human, SP or hotseat (< 3 players) ═══
    elif player_count < 3:
        if is_human(civA) AND civA == current_player:
            if contactExpired: open_diplomacy_screen(civA, civB, ...)
        if is_human(civB) AND civB == current_player:
            if contactExpired: open_diplomacy_screen(civB, civA, ...)

    // ═══ CASE 3: Both human, true multiplayer (≥ 3 players) ═══
    else:
        if contactExpired:
            // ── Multiplayer parley handshake ──
            DAT_0063f278 = -1              // parley response code
            DAT_00626a2c = 1               // diplomacy in progress

            // Send parley request to remote player
            send_mp_event(0x3D, civB, civA)
            show_dialog("PARLEYWAITING")   // blocking wait

            if mp_abort:
                show_message("PARLEYBUSY")
                DAT_00626a2c = 0
            elif parley_cancelled:
                show_message("PARLEYCANCEL")
                DAT_00626a2c = 0
            elif parley_timeout:
                send_mp_event(0x81, civB)  // timeout
                DAT_0067a8c0 = -1          // clear target
                DAT_00626a2c = 0
            elif parley_refused:
                show_message("PARLEYGOAWAY")
                DAT_00626a2c = 0
            elif parley_accepted:
                DAT_0067a8c0 = civB        // set target
                show_message("PARLEYOK")
                if not mp_abort AND not cancelled:
                    start_mp_negotiation(civB, 3)
                else:
                    show_message("PARLEYCANCEL" or "PARLEYBUSY")
                    DAT_00626a2c = 0

    // ═══ POST-CONTACT: First contact handling ═══
    if (treaty[civA][civB] & 1) == 0:
        set_treaty_flag(civA, civB, 0x4000)   // recent contact flag

    wasFirstContact = (treaty[civA][civB] & 1) == 0
    set_treaty_flag(civA, civB, 0x401)         // contact + ceasefire

    // ═══ AI-vs-AI DIPLOMACY ═══
    if not is_human(civA) AND not is_human(civB)
       AND ((game_turn + civA + civB) % 4 == 0 OR wasFirstContact)
       AND not (scenario blocks):

        // ── Evaluate war ──
        warA = ai_should_declare_war(civA, civB)
        warB = ai_should_declare_war(civB, civA)

        // Peace if neither wants war, or blocked by wonder/govt
        peaceBlocked_A = civ_has_wonder(civB, 6) OR civ_has_wonder(civB, 0x18)
                         OR (civ[civA].govt > 4 AND not (treaty_byte1[civA][civB] & 0x08))
        peaceBlocked_B = civ_has_wonder(civA, 6) OR civ_has_wonder(civA, 0x18)
                         OR (civ[civB].govt > 4 AND not (treaty_byte1[civB][civA] & 0x08))

        if (warA == 0 OR peaceBlocked_A) AND (warB == 0 OR peaceBlocked_B):
            // ── Peace path ──
            ai_tech_exchange(civA, civB)

            if not (treaty[civA][civB] & 0x04):     // no peace treaty yet
                // Show peace messages (UN force, Great Wall, etc.)
                if treaty_byte1[civA][civB] & 0x08:
                    if civ_has_wonder(civB, 0x18): show("UNFORCE")
                    elif civ_has_wonder(civB, 6): show("WALLFORCE")
                // ... same for civB
                // Notify allied human players
                if allied_with_human(civA): show("ALLYMAKESPEACE")
                elif allied_with_human(civB): show("ALLYMAKESPEACE")
                elif has_intel_access: show("SIGNPEACE")
                set_treaty_flag(civA, civB, 0x04)    // peace treaty

            else:
                // ── Alliance evaluation ──
                existingAlliance = treaty[civA][civB] & 0x08
                willFormAlliance = existingAlliance

                // Superpower check: rank 7 + rank > 3 → won't ally
                if civ[civA].powerRank == 7 AND civ[civB].powerRank > 3:
                    willFormAlliance = false
                if civ[civB].powerRank == 7 AND civ[civA].powerRank > 3:
                    willFormAlliance = false

                // Search for mutual enemy to ally against
                mutualEnemy = -1
                for thirdCiv = 1 to 7:
                    if thirdCiv == civA OR thirdCiv == civB: continue
                    // Complex alliance formation logic...
                    // Find a third civ that both dislike or are at war with
                    // Check superpower dominance scenarios
                    if suitable_enemy_found:
                        mutualEnemy = thirdCiv
                        break

                // Scenario special case: civs 6 & 7 always ally after turn 1
                if scenario_mode AND ((civA==6 AND civB==7) OR (civA==7 AND civB==6)):
                    if game_turn > 1: mutualEnemy = 0

                if mutualEnemy < 1 OR (conditions_against_alliance):
                    // ── Break existing alliance ──
                    if existingAlliance AND not ai_autoplay AND has_intel_access:
                        show("CANCELALLIED")
                    clear_treaty_flag(civA, civB, 0x08)

                else:
                    // ── Form/maintain alliance ──
                    if not existingAlliance AND has_intel_access AND not ai_autoplay:
                        if is_superpower_scenario:
                            show("SIGNNATO")       // NATO-style pact
                        else:
                            show("SIGNALLIED")     // regular alliance

                    // Try to drag mutualEnemy into war
                    joined = check_join_war(civA, mutualEnemy, civB)
                    joined |= check_join_war(civB, mutualEnemy, civA)

                    // If mutualEnemy is human + not at war + no joining:
                    // set "city capture" flag (0x10000) as provocation
                    if is_human(mutualEnemy) AND not at_war(mutualEnemy)
                       AND joined == 0:
                        if not (treaty_byte1[civA][mutualEnemy] & 0x20):
                            set_treaty_flag(civA, mutualEnemy, 0x10000)
                        if not (treaty_byte1[civB][mutualEnemy] & 0x20):
                            set_treaty_flag(civB, mutualEnemy, 0x10000)

                    set_treaty_flag(civA, civB, 0x08)   // alliance

        else:
            // ── War path ──
            if not (treaty_byte1[civA][civB] & 0x20) OR wasFirstContact:
                // Ally notification: "ALLYMAKESWAR"
                if allied_with_human(civA):
                    if ai_should_declare_war(civA, civB) == 0:
                        notify_ally_betrayal(civA, civB)
                    show("ALLYMAKESWAR", civA, civB)
                elif allied_with_human(civB):
                    if ai_should_declare_war(civB, civA) == 0:
                        notify_ally_betrayal(civB, civA)
                    show("ALLYMAKESWAR", civB, civA)
                elif has_intel_access:
                    // Show "DECLAREWAR" or "CANCELPEACE"
                    if not (treaty[civA][civB] & 0x04) OR (treaty[civA][civB] & 0x08):
                        show("DECLAREWAR")
                    else:
                        show("CANCELPEACE")

                // Declare war
                clear_treaty_flag(civA, civB, 0x04)     // break peace
                set_treaty_flag(civA, civB, 0x2000)     // vendetta

                // If had embassy, break it + set betrayal flag
                if treaty[civA][civB] & 0x10:
                    clear_treaty_flag(civA, civB, 0x10)
                    treaty[civA][civB] |= 0x800         // betrayal flag
```

function ai_choose_government(civIdx, isRevolution):
    // ── Scenario override ──
    if (game_flags & 0x80) AND (scenario_flags & 0x10)
       AND civ[civIdx].government != 0:
        return                                     // scenario locks government

    // ── Determine max government level ──
    maxGovt = 6                                    // default: up to Democracy
    if isRevolution:
        maxGovt = 5                                // during revolution: cap lower
        if rand() % 3 != 0:
            maxGovt = 4                            // 2/3 chance: even lower

    // ── Tech deficit auto-demotion ──
    if (game_flags & 0x01)                         // standard game (not scenario)
       AND civ[civIdx].powerRank < 6:              // not a superpower

        leadTechLevel = civ[strongest_civ].techLevel
        ourTechLevel = civ[civIdx].techLevel

        if leadTechLevel - ourTechLevel > 6:
            civ[civIdx].govtPreference[0] = -2     // DAT_0064ca80: strong demotion signal

        if leadTechLevel - ourTechLevel > 8:
            civ[civIdx].govtPreference[-1] = -1    // DAT_0064ca7e: extreme demotion

    // ── If AI has positive preference for despotism ──
    if civ[civIdx].govtRanking[0] > 0:             // DAT_0064ca74
        maxGovt = 3                                // limit to early govts
        if civ[civIdx].government < 6:
            maxGovt = 1                            // already non-democracy → stay low

    // ── Find best available government ──
    bestScore = -999
    bestGovt = 1                                   // default: Despotism
    for g = 1 to maxGovt:
        if check_govt_available(civIdx, g):
            ranking = civ[civIdx].govtRanking[g]   // DAT_0064ca74 + g*2
            if ranking >= bestScore:
                bestScore = ranking
                bestGovt = g

    // Apply
    ai_revolution_notification(civIdx, bestGovt)

function ai_military_aid(civIdx):
    // Count how many civs are at war with civIdx
    warCount = 0
    for other = 1 to 7:
        if other != civIdx AND (treaty_byte1[other][civIdx] & 0x20):
            warCount += 1

    // ── Find an ally that needs help ──
    for allyCiv = 1 to 7:
        if allyCiv == civIdx: continue
        if not (treaty[allyCiv][civIdx] & 0x08): continue     // not allied

        // Ally must be weaker or we have no wars of our own
        if civ[allyCiv].totalAtkDefSum > civ[civIdx].totalAtkDefSum
           AND warCount > 0:
            continue

        // ── Find enemy that ally is fighting ──
        for enemyCiv = 1 to 7:
            if enemyCiv == civIdx OR enemyCiv == allyCiv: continue
            if not (treaty_byte1[enemyCiv][allyCiv] & 0x20): continue  // not at war
            if not (treaty_byte1[enemyCiv][allyCiv] & 0x02): continue  // no ceasefire broken
            if civ[allyCiv].totalAtkDefSum > civ[enemyCiv].totalAtkDefSum: continue
            if civ[allyCiv].powerRank > civ[enemyCiv].powerRank: continue

            // ── Find best unit to gift ──
            bestUnit = -1
            bestScore = 0

            for unitSlot = 0 to max_units - 1:
                if unit[unitSlot].alive == 0: continue
                if unit[unitSlot].owner != civIdx: continue

                unitType = unit[unitSlot].typeId
                domain = unit_types[unitType].domain    // must be land (< 2)
                if domain >= 2: continue
                if unit_types[unitType].civilian: continue   // not civilian

                ux = unit[unitSlot].x
                uy = unit[unitSlot].y

                // Unit must be in a city with other units
                if not is_tile_valid(ux, uy): continue
                if get_city_owner_at(ux, uy) < 0: continue
                if count_units_at(unitSlot, 2) <= 1: continue  // alone

                // Must be on a continent with AI goal = attack(4) or defend(5)
                continent = get_body_id(ux, uy)
                goal = civ[civIdx].continent_status[continent]
                if goal != 4 AND (goal != 5 OR warCount > 0): continue

                // Score = (defense + attack*2) * hp_percentage
                attack = unit_types[unitType].attack
                defense = unit_types[unitType].defense
                hpPct = get_unit_hp_pct(unitSlot)
                score = (defense + attack * 2) * hpPct
                if score > bestScore:
                    bestUnit = unitSlot
                    bestScore = score

            if bestUnit < 0: continue

            // ── Find destination city (enemy city that ally is attacking) ──
            for cityIdx = 0 to max_cities - 1:
                if city[cityIdx].alive == 0: continue
                if city[cityIdx].owner != allyCiv: continue

                cx = city[cityIdx].x
                cy = city[cityIdx].y
                enemyPresence = get_unit_owner_at(cx, cy)
                if enemyPresence < 0: continue

                // City must be under attack, on same continent as goal
                if not (city[cityIdx].flags & 0x80): continue
                continent = get_body_id(cx, cy)
                if civ[allyCiv].continent_status[continent] != 1: continue
                if has_enemies_nearby(cx, cy, allyCiv): continue

                // Check pathfinding
                pathCity = find_nearest_city(bestUnit.x, bestUnit.y)
                if pathCity < 0: continue

                // ── Transfer unit ──
                pick_up_unit(bestUnit, 1)

                // Update per-civ unit type counts
                civ[civIdx].unitCounts[bestUnit.typeId] -= 1
                unit[bestUnit].owner = allyCiv
                unit[bestUnit].homeCityId = cityIdx
                unit[bestUnit].orders = 0              // clear orders
                civ[allyCiv].unitCounts[bestUnit.typeId] += 1

                // Move unit to destination city
                place_unit_at(bestUnit, cx, cy, 1)

                // Refresh both cities
                citywin_refresh(cityIdx)
                redraw_tile(city[cityIdx].x, city[cityIdx].y)
                citywin_refresh(pathCity)
                redraw_tile(city[pathCity].x, city[pathCity].y)

                // ── Notify human players ──
                if is_human(enemyCiv) AND not ai_autoplay:
                    // "MILITARYAID1" — from ally's perspective
                    set_params(civIdx_name, unit_type_name, allyCiv_name)
                    if current_player == enemyCiv:
                        show_message("MILITARYAID1", bestUnit)
                    elif multiplayer:
                        send_mp_event(0x51, enemyCiv, bestUnit)

                if is_human(allyCiv) AND not ai_autoplay:
                    // "MILITARYAID2" — from recipient's perspective
                    set_params(civIdx_name, unit_type_name, city_name, enemyCiv_name)
                    if current_player == allyCiv:
                        show_message("MILITARYAID2", bestUnit)
                    elif multiplayer:
                        send_mp_event(0x52, allyCiv, bestUnit)

                // MP sync
                if multiplayer:
                    send_packet(0xFF, 2)
                    flush_send_buffer(5000)
                return
```
