function evaluate_city_tiles(city, city_index):
    // Calculate total food, shields, trade for all worked tiles
    total_food = 0
    total_shields = 0
    total_trade = 0

    // City center tile always worked (offset 0 maps to city gx,gy)
    for offset = 0 to 20:
        if get_worker_tile_status(city, offset) != WORKED:

function adjust_specialist_count(city):
    // Ensure specialists + workers == city.size
    total_workers = count worked tiles (excluding city center)
    total_specialists = city.elvis + city.taxman + city.scientist
    expected = city.size - 1  // -1 for city center
    if total_workers + total_specialists > expected:
        // Remove excess specialists (scientist first, then taxman, then elvis)
