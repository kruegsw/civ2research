function recalc_viewport_geometry(this):
    // Save client area dimensions
    this->+0x328 = this->+0x12C     // client_width
    this->+0x32C = this->+0x130     // client_height

    // Compute tile dimensions from zoom
    tw = scale_sprite(0x40, zoom)    // tile_width = (zoom+8)*64/8
    th = scale_sprite(0x20, zoom)    // tile_height = (zoom+8)*32/8
    half_tw = tw / 2
    half_th = th / 2
    quarter_tw = tw / 4
    quarter_th = th / 4

    // Stride = full tile width, row height = tile_height + half_tile_height
    this->+0x320 = tw
    this->+0x324 = th + half_th

    // Viewport tile counts
    cols_narrow = (client_w - 1 + tw) / tw       // +0x2F8
    rows = (client_h - 1 + th) / th              // +0x2FC
    cols_wide = (half_tw*3 - 1 + client_w) / tw  // +0x300
    rows_wide = (half_th*3 - 1 + client_h) / th  // +0x304

    // Clamp to map dimensions
    margin_x = 0; margin_y = 0
    if cols_narrow > map_width/2:
        cols_narrow = map_width/2
        margin_x = (client_w - tw * cols_narrow) / 2
        cols_wide = cols_narrow + 1
    if rows > map_height:
        rows = map_height/2
        margin_y = (client_h - th * rows) / 2
        rows_wide = map_height + 1

    // Apply wrapping and parity constraints
    wrap_x(cursor_x)
    viewport_top_y = cursor_y - rows
    viewport_left_x = wrap(cursor_x - cols_narrow)

    // GAME CONSTANT: flat earth check
    if (DAT_00655AE8 & 0x8000) != 0:    // flat earth
        clamp viewport_left_x to [0, map_width - 2*cols_narrow]
    clamp viewport_top_y to [0, map_height - 2*rows]

    // Ensure even parity for isometric grid
    if viewport_left_x & 1: viewport_left_x -= 1
    if viewport_top_y & 1: viewport_left_x += 1

    // Compute viewport end coords
    viewport_right_x = wrap(viewport_left_x + cols_wide*2 - 2)
    viewport_bottom_y = viewport_top_y + rows_wide*2 - 2

    // Rebuild font sizes if zoom changed
    if zoom != last_zoom:
        resize_cursor_bitmaps(tw, th)
        last_zoom = zoom
        set_sprite_scale(zoom)
        rebuild_cursor_sprites()
        reset_sprite_scale()

    // Font sizes proportional to tile height
    font_large = (th * 3) / 5      // city name font
    font_small = th / 4             // unit count font
```

> **Game logic note**: Viewport geometry follows standard isometric engine: odd/even row offset, diamond-grid coordinate system. The `& 0x8000` check on DAT_00655AE8 is the "flat earth" flag — wrapping maps don't clamp x. The JS renderer uses the same approach (see `recalcViewport` in app.js).

function calc_coast_quadrants(tile_x, tile_y):
    // GAME CONSTANTS: coastline quadrant bitmask system
    // DAT_0066C720[4] = quadrant flags: bits 1=side, 2=diagonal, 4=corner
    // Quadrants: 0=NE, 1=SE, 2=SW, 3=NW
    clear DAT_0066C720[0..3]
    if !is_tile_valid(tile_x, tile_y): return 0

    neighbor_count = 0
    for dir = 0 to 7:   // 8 directions
        nx = wrap(tile_x + DAT_00628350[dir])   // dx offsets
        ny = tile_y + DAT_00628360[dir]          // dy offsets
        if !is_tile_valid(nx, ny):
            terrain_type = 7    // treat OOB as land
        else:
            terrain_type = get_terrain(nx, ny)

        if terrain_type != 10:   // not ocean = land neighbor
            neighbor_count++
            if dir is even (cardinal):
                // Side bit: affects two quadrants
                q1 = dir >> 1
                q2 = (q1 + 1) & 3
                DAT_0066C720[q1] |= 4    // corner bit
                DAT_0066C720[q2] |= 1    // side bit
            else:
                // Diagonal: affects one quadrant
                q = ((dir + 1) & 6) >> 1
                DAT_0066C720[q] |= 2     // diagonal bit
    return neighbor_count
```

> **Game logic note**: The coastline system uses 3 bits per quadrant (side=1, diagonal=2, corner=4) to select from 7 coast sprite variants per quadrant. Terrain type 10 = ocean, 7 = default land for out-of-bounds. This exactly matches our JS `_calcCoastQuadrants()` in renderer.js.
