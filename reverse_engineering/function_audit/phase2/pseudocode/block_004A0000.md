```
function open_text_file_section(filename, sectionName):
    clear stringBuf, headerBuf
    if sectionName != NULL:
        headerBuf = "@" + strupr(sectionName)
    if filename == NULL:
        alreadyOpen = true
    else:
        close_text_file()
        copy filename to pathBuf
        save cwd, chdir to game dir
        textFileHandle = fopen(pathBuf, "r")
        if textFileHandle and strlen(pathBuf) > 0:
            cachedOffset = find_section_offset(pathBuf, headerBuf)
        chdir to alt dir
        if textFileHandle == NULL:
            textFileHandle = fopen(pathBuf, "r")  // try alternate path
            if opened: cachedOffset = find_section_offset(...)
        restore cwd
        if textFileHandle == NULL: goto fail
    if sectionName == NULL:
        return 0  // success, no section seek needed
    if cachedOffset > 0:
        fseek(textFileHandle, cachedOffset, SEEK_SET)
    loop:
        if EOF:
            if alreadyOpen: rewind and retry once
            else: goto fail
        line = fgets(lineBuf, 255)
        strip_whitespace(line)
        strip_comments(line)
        if strcmpi(line, headerBuf) == 0:
            found = true; break
    tokenPtr = lineBuf + strlen(lineBuf)
    return 0  // success
fail:
    close_text_file()
    return 1  // error
```

```
function calc_civ_score(civId):
    // --- Population score ---
    populationScore = 0
    for each city c where city.exists and city.owner == civId:
        populationScore += city.size + city.happyCitizens - city.unhappyCitizens

    // --- Wonder score ---
    wonderScore = 0
    for w in 0..27:
        if wonder[w].cityId >= 0 and wonder[w].city.owner == civId:
            wonderScore += 20

    // --- Spaceship score ---
    spaceshipScore = 0
    peacePenalty = 0
    call calc_spaceship_stats(civId, 1)
    if SPACESHIP_MULTIPLIER != 0 and civ[civId].spaceshipPopulation != 0:
        spaceshipScore = civ[civId].spaceshipPopulation * SPACESHIP_MULTIPLIER
        if civ[civId].attribs & 0x10:  // spaceship arrived
            spaceshipScore = 0
            peacePenalty = civ[civId].spaceshipPopulation * SPACESHIP_MULTIPLIER

    // --- Pollution penalty ---
    pollutionPenalty = (globalPollution - cleanedPollution) * -10

    // --- Map exploration bonus ---
    mapBonus = 0
    if turn > 199:
        mapBonus = clamp(mapExploredPercent * 3, 0, 100)

    // --- Future tech bonus ---
    futureTechScore = civ[civId].futureTechCount * 5

    // --- Difficulty modifier ---
    difficultyMod = difficultyLevel * 25 - 50  // range: -50 to +75

    // --- Base score ---
    baseScore = mapBonus + pollutionPenalty + peacePenalty + wonderScore +
                populationScore + difficultyMod + futureTechScore
    if baseScore < 0: baseScore = 0

    // --- Spaceship arrival score (alternate) ---
    arrivalScore = 0
    if only this civ is human:
        yearFactor = turn
        if RETIREMENT_FLAG and yearFactor < 251: yearFactor = 250
        arrivalScore = spaceshipMult * 100 + (570 - yearFactor) * 2 + 400
        if SCENARIO_FLAG_4: arrivalScore = arrivalScore * 5 / 4
        if SCENARIO_FLAG_8: arrivalScore = arrivalScore * 4 / 5

    // --- Alpha Centauri relative-power ranking ---
    if ALPHA_CENTAURI_FLAG and ALPHA_CENTAURI_MODE:
        totalPop = sum of all cities' population
        leaderPop = population of scenario leader civ
        civPop = population of civId's cities
        if scenario leader is barbarian: leaderPop = civPop
        rank = compute_rank(civPop vs leaderPop thresholds)
        // rank 0=supreme (1000), 1=great (750), 2=good (500), 3=ok (250), 4=poor (0)
        if not leader and leader != civId: rank = 4 - rank  // invert for non-leaders
        relativeScore = rank_bonus + civPop * 10
        if arrivalScore > 0: relativeScore *= 2
        arrivalScore = -1  // suppress normal arrival score
        baseScore = relativeScore
        return relativeScore
    else:
        return max(baseScore, arrivalScore)
```

```
    listIndex = buttonId - 0x40F
    get_mouse_position(mouseX, mouseY)
    if itemCount > 0:
        clickedItem = hit_test_listbox(mouseX, mouseY, listIndex)
        if clickedItem valid and clickedItem < itemCount:
            if shift_key_down:
                // Range select: select all between last click and current
                for i in min(last, clicked)..max(last, clicked):
                    selected[i] = 1
            elif ctrl_key_down:

                lastClicked = clicked
```

```
function hit_test_listbox(mouseX, mouseY, listIndex):
    dialog = get_dialog_this()
    if mouseY < rect.top: return -1  // above
    if mouseY >= rect.bottom: return -2  // below
    if mouseX < rect.left: return -3  // left
    if mouseX >= rect.right: return -4  // right
    // Calculate item index from Y position
    itemHeight = rect_height / visibleItems
    index = (mouseY - rect.top) / itemHeight + scrollOffset
    return index
```
