## FUN_005746a1 / sprited_get_sprite_coords
**Category**: RN.sprite | **Size**: 1362b | **Ported**: NN

```
// Maps sprite index to sheet coordinates; returns total sprite count
switch (editor_type):
    case 0:  // ICONS small (4x5 grid)
        count=20; w=0x24; h=0x14
        x = (idx%4)*0x25 + 0x157;  y = (idx/4)*0x15 + 0xD3
    case 1:  // CITIES (2 banks of 24, 4-col layout)
        count=48; w=0x40; h=0x30
        bank = idx / 24;  slot = idx % 24
        x = (bank==0 ? 1 : 0x14E) + (slot%4)*0x41
        y = (slot/4)*0x31 + 0x27
    case 2:  // CITIES alt (4 slots)
        count=4; w=0x40; h=0x30
        x = idx*0x41 + 0x8F;  y = 0x1A7
    case 3:  // Small icons (9x2 grid)
        count=18; w=0x0E; h=0x16
        x = ((idx/2)%9) * (w+1) + 1;  y = (idx&1) * (h+1) + 0x1A9
    case 4:  // ICONS extended (38 + 28)
        count=66; w=0x24; h=0x14
        if idx < 38: x = (idx%8)*0x25+0x157;  y = (idx/8)*0x15+1
        else: x = ((idx-38)%7)*0x25+0x157;  y = ((idx-38)/7)*0x15+0x6A
    case 5:  // Special (1 sprite)
        count=1; w=0x40; h=0x20; x=199; y=0x100
    case 6:  // Misc (8 sprites)
        count=8; w=0x20; h=0x20; x = (w+1)*idx+1; y=0x164
    case 7:  // TERRAIN1 base (44 sprites: 11 types x 4 variants)
        count=44; w=0x40; h=0x20
        x = (idx/11)*0x41+1;  y = (idx%11)*0x21+1
    case 8:  // TERRAIN2 coast (64+4=68)
        count=68; w=0x40; h=0x20
        if idx < 64:
            x = (idx&7)*0x41+1
            y = 0x43 + (idx/16)*0x42 + ((idx&15)>=8 ? 0x21 : 0)
        else:
            x = (idx-64)*0x41+1;  y = 0x14B
    case 9:  // TERRAIN2 dither (8x4 = 32)
        count=32; w=0x20; h=0x10; x=1; y=0x1AD
        sub = idx % 4  // 4 sub-positions with different y offsets
        x += (idx/4) * 0x42
    case 10: // TERRAIN1 extra (6+18 = 24)
        count=24; w=0x40; h=0x20
        if idx < 6: x=0x1C8; y=idx*0x21+100
        else: x = ((idx-6)%9)*0x41+1; y = ((idx-6)/9)*0x21+0x16C
    case 11: // UNITS (9-col grid)
        count=63; w=0x40; h=0x30
        x = (idx%9)*0x41+1;  y = (idx/9)*0x31+1
return count
```
