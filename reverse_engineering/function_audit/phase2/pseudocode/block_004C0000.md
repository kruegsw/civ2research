### Pseudocode

```
result = false

// Check scenario flags: if scenario bit 0x80 set AND rules bit 0x04 set
// AND not OCC mode AND not (special bit AND wonderIndex==0x17=Apollo),
// then wonders can't be built
if (scenarioFlags & 0x80) != 0 AND (rulesFlags & 4) != 0
   AND occMode == 0
   AND NOT ((scenarioFlags & 0x8000) != 0 AND wonderIndex == 0x17):
    return false

// Check wonder slot availability (wonderCityIds[wonderIndex] == -1)
if wonderCityIds[wonderIndex] != -1:
    return false

// Check tech prereq: improvement table entry at index (wonderIndex + 0x27)
// DAT_0064c48e = improvement_table[].prereqTech
prereqTech = improvement_table[(wonderIndex + 0x27)].prereqTech
if NOT civ_has_tech(civSlot, prereqTech):
    return false

result = true
return result
```

```
// Coastal buildings (Harbour=0x1e, Offshore Platform=0x1f): need coastal bit
if buildingId == 0x1e OR buildingId == 0x1f:
    if (city[cityIndex].attribs1 & 0x80) == 0:  // not coastal
        return false

// Port Facility=0x22, Coastal Fortress=0x1c: need river/coastal AND coastal
if buildingId == 0x22 OR buildingId == 0x1c:
    if (city[cityIndex].attribs2 & 0x20) == 0:   // no river
        return false
    if (city[cityIndex].attribs1 & 0x80) == 0:   // not coastal

// Police Station=0xb: only if no other exists
if buildingId == 0xb:
    if wonder_exists(civSlot, 10):
        return false

// Granary=7: can't have if city already has Aqueduct=1
if buildingId == 7:
    if city_has_building(cityIndex, 1):
        return false

// Hydro Plant=10: requires Factory=5 AND hydro tech
if buildingId == 10:
    if NOT city_has_building(cityIndex, 5):
        return false
    if NOT civ_has_tech(civSlot, hydro_tech):
        return false

// Recycling Center=0x16: requires Factory=10 tech
if buildingId == 0x16:
    if NOT city_has_building(cityIndex, 10):
        return false

// Power Plant exclusion=0xc: needs Factory=6
if buildingId == 0xc:
    if NOT city_has_building(cityIndex, 6):
        return false

// Solar Plant=0x1a: no existing solar, needs Recycling=0xc AND solar tech
if buildingId == 0x1a:
    if wonder_exists(civSlot, 0x1a):
        return false
    if NOT city_has_building(cityIndex, 0xc):
        return false

// Supermarket=0x17: needs Aqueduct=9
if buildingId == 0x17:
    if NOT city_has_building(cityIndex, 9):
        return false

// Research Lab=0x21: no existing lab wonder
if buildingId == 0x21:
    if wonder_exists(civSlot, 0x15):
        return false

// MFG Plant=0x10: needs Factory=0xf AND mfg tech
if buildingId == 0x10:
    if NOT city_has_building(cityIndex, 0xf):
        return false

// Mass Transit=0x14: needs Railroad bit
if buildingId == 0x14:
    if (city[cityIndex].attribs1_byte1 & 8) == 0:
        return false

// Power plant mutual exclusion (0x13=Nuclear, 0x14=Solar, 0x1d=Hydro, 0x15=Wind):
// Must have Factory(0xf) OR MFG Plant(0x10)
// Must not already have a competing power plant type (unless replacing)
if buildingId in {0x13, 0x14, 0x1d, 0x15}:

       AND NOT city_has_building(cityIndex, 0x10):
        return false
    // Complex exclusion: no other power plant if one exists
    // (Nuclear blocks Solar blocks Hydro etc.)
    if buildingId != 0x1d:
        if wonder_exists(civSlot, 0x16):
            return false
        if city_has_building(cityIndex, 0x14):
            return false
        if city_has_building(cityIndex, 0x15) AND buildingId != 0x14:
            return false
        if city_has_building(cityIndex, 0x1d):
            return false

// Spaceship parts (0x23-0x25): special validation
if buildingId >= 0x23 AND buildingId <= 0x25:
    if (flags & 0x80) != 0:
        return false
    if spaceshipLaunched:
        return false
    if turnNumber <= 0 AND (scenarioFlags & 0x80) == 0:
        return false
    if alpha_centauri_already_reached(civSlot):
        return false

return true
```

```c
int ai_pick_research_goal(int civSlot)
// Returns: tech ID to research, or -1 if none available
```

### Pseudocode
```
bestTech = -1
bestScore = -1
availableCount = 0

for techId = 0 to 99:
    if NOT canResearch(civSlot, techId):
        continue
    // Both prereqs must not be (-2, -2) = unresearchable
    if prereq1[techId] == -2 AND prereq2[techId] == -2:
        continue

    // Human player: skip every 3rd tech on non-chieftain
    isHuman = (human_players_bitmask & (1 << civSlot)) != 0
    if isHuman AND availableCount > 0 AND difficulty != chieftain:
        if (techId - civPersonaIndex[civSlot]) % 3 == 0:
            continue

    availableCount++

    // Score calculation differs for AI vs human
    if NOT isHuman:  // AI
        techValue = calcTechValue(civSlot, techId)
        if techValue <= 1:
            score = 0
        else:
            score = rand() % techValue
    else:  // Human
        techValue = calcTechValue(civSlot, techId)
        score = rand() % 3 + techValue - 1

    if score > bestScore:
        bestScore = score
        bestTech = techId

return bestTech
```

## FUN_004c2788 / calc_tech_cost
**Category**: GL.research | **Size**: 1003b | **Ported**: PA

### Signature
```c
int calc_tech_cost(int civSlot)
// Returns: beakers needed to complete current research
```

### Pseudocode
```
// Total known techs (both discovered + given)
totalTechs = civData[civSlot].techCount + civData[civSlot].futureTechCount
if totalTechs < 2:
    totalTechs = 1

// Difficulty-scaled base cost
difficultyIdx = clamp(difficulty, 0, 4)
if isHuman:
    baseCost = difficultyIdx * 2 + 6
else:  // AI
    baseCost = 14 - difficultyIdx

// ── Tech leader comparison (AI catch-up / human penalty) ──
if NOT scenario OR scenarioCatchUp == 0:
    leaderTechs = techCount[techLeaderCiv] + futureTechCount[techLeaderCiv]
    if totalTechs < leaderTechs:
        // Behind tech leader: reduce cost
        if difficulty > 0:
            baseCost -= 1
        if difficulty == 5 AND totalTechs + 4 < leaderTechs AND turn > 150:
            baseCost -= 1  // extra Deity catch-up
    else:
        // Ahead of leader: increase cost proportionally
        baseCost += (totalTechs - leaderTechs) / 3

    // Late-game penalty (totalTechs > 19)
    latePenalty = 0
    if totalTechs > 19:
        latePenalty = clamp(totalTechs - turn/8, 0, 6)
    baseCost += latePenalty

// ── COSMIC multiplier (scenario/rules override) ──
if NOT scenario:
    if cosmicTechMultiplier != 10:  // DAT_0064bcd3
        baseCost = (cosmicTechMultiplier * baseCost) / 10
else:
    if scenarioTechMultiplier != 10:  // DAT_0064bcb2
        baseCost = (scenarioTechMultiplier * baseCost) / 10

// Progressive modifier (75% adder, scaled for early game)
modifier = (baseCost * 3) >> 2  // 75% of baseCost
if totalTechs < 20:
    modifier = (totalTechs * modifier) / 20
baseCost += modifier

// ── World size penalty ──
if maxCities > 67:  // DAT_00655b1a
    baseCost = (baseCost * 67) / maxCities

// ── Game option flags ──
if (gameFlags & 4) != 0:  // "Raging hordes" or similar
    baseCost = (baseCost * 5 + 3) / 4  // multiply by 1.25
if (gameFlags & 8) != 0:  // another option
    baseCost = (baseCost * 4) / 5  // multiply by 0.8

// Human minimum floor
if isHuman AND baseCost < (11 - totalTechs):
    baseCost = 11 - totalTechs

// Final cost = baseCost * totalTechs, clamped to [1, 32000]
cost = baseCost * totalTechs
if cost < 1 OR cost > 32000:
    cost = 32000
return cost
```

### Key References
- **Calls**: thunk_FUN_005adfa0 (clamp)
- **Called by**: FUN_004c2b73 (add_research_beakers)
- **Globals**: DAT_0064c6b0 (tech count per civ), DAT_0064c6b2 (future tech count), DAT_00655b08 (difficulty), DAT_00655b0b (human players), DAT_00655af8 (turn), DAT_00655c20 (tech leader civ), DAT_0064bcd3 (cosmic tech multiplier), DAT_0064bcb2 (scenario tech multiplier), DAT_00655af0 (scenario/game flags), DAT_00655b1a (max cities)

### Porting Notes

The JS version implements the human formula but **missing**: AI cost formula (14 - difficulty), tech leader comparison (catch-up for AI behind, penalty for AI ahead), Deity extra catch-up, world size penalty (max cities > 67 scaling), and the `gameFlags & 8` modifier. // Returns: 0 = spy survives, 1 = spy killed

### Pseudocode
```
owner = unit[unitIndex].owner

// On successful action: restore movement points
if successLevel >= 0:
    unit.movesRemaining += moveMultiplier

// ── Spy (type 0x2F='/') survival check ──
if unit[unitIndex].type == 0x2F:  // spy (as opposed to diplomat)
    survivalOdds = (successLevel < 0) + 2   // base: 2 or 3
    if unit has veteran flag (0x2000):
        survivalOdds *= 2
    if successLevel > 0:
        survivalOdds /= 2

    // Random survival roll
    if survivalOdds < 2:
        roll = rand() % 2  // coin flip bonus
        if roll: survivalOdds++

    if survivalOdds <= 1:
        escapeChance = 0
    else:
        escapeChance = rand() % survivalOdds

    if escapeChance != 0:
        // Spy survives! Relocate to nearest friendly city
        cityX, cityY = find_nearest_city(unit.x, unit.y, owner, -1, -1)
        if cityIdx >= 0:
            teleport_unit(unitIndex, city.x, city.y, 1)
            release_unit(unitIndex)
            refresh_tiles(oldPos, newPos)
            // MP notification
            if MP_players > 2: send_move_notification(...)

        // Show success dialog
        if isHuman AND NOT multiplayer:
            if NOT veteran:
                show_dialog("BOND007", unit_sprite)
            else:
                show_dialog("BOND007_VET", unit_sprite)

        // Set veteran flag
        unit.flags |= 0x2000

        return 0  // spy survived

    // Spy caught
    if isHuman AND NOT multiplayer AND successLevel >= 0:
        show_dialog("BONDGLORY", unit_sprite)

// ── Spy/diplomat dies ──
kill_unit(unitIndex, 1)

// Trigger diplomatic incident if target specified
if targetCiv > 0 AND successLevel >= 0:
    handle_incident_terror(owner, targetCiv)

return 1
```

## FUN_004c66ba / execute_civil_war

```
oldOwner = city[cityIndex].owner
cityX = city[cityIndex].x
cityY = city[cityIndex].y

    // Grant visibility around city to new owner (21 tiles)
    save_undo()
    for each of 21 tiles around city:
        if tile in bounds:
            grant_visibility(tile, 1 << newOwnerCiv)
            update_vis(tile, newOwnerCiv)
    restore_undo()

    // Show notification to relevant players
    if NOT multiplayer:

            if isHuman(oldOwner) AND MP_players > 2:
                send_notification(0x5e, ...)

    unitIdx = maxUnits - 1  // iterate backwards
    while unitIdx >= 0:

            dist = calc_iso_distance(unit.x, unit.y, cityX, cityY)
            if dist < 2:

                    // Update per-civ unit-type counts

                    grant_unit_visibility(unit.x, unit.y, newOwnerCiv, 1)

    transfer_city(cityIndex, newOwnerCiv, revolterCount)
```
