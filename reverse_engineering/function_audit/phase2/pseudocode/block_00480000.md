function mp_relink_event_strings():
    // Rebuilds serialized event string pool after network transfer
    // Event records: stride = 0x1C4 (452 bytes), linked list via +0x1BC/+0x1C0
    if event_count == 0: return

    cursor = event_string_base
    event_ptr = event_string_base + event_index * 0x1C4

    // Phase 1: Clear link pointers, build backward-linked list
    for i in 0..event_index:
        event_ptr.next_ptr = 0   // +0x1BC..+0x1C3
        if i > 0:
            event_ptr.prev = event_ptr - 0x1C4
            event_ptr.prev.next = event_ptr

    // For each non-null string pointer at known offsets, advance cursor
    // by max(strlen+1, min_size) where min_size varies by field:

function mp_handle_disconnections() -> bool:
    if not dedicated_server:
        // Client-side check
        if global_disconnect_counter != 0:
            game_active = 0
            log("We lost connection to the server")
            disconnect(1)
            show_dialog("LOSTSERVER")
            return true  // fatal
    else:
        // Server-side: purge timed-out players
        purged = purge_disconnected_players()

        // Check for civs with no connected player slot
        for civ = 1..7:
            if civ_alive AND civ_is_human:
                found = false
                for slot in 0..6:
                    if slot.active AND slot.civ == civ: found = true; break
                if not found:
                    human_civs_bitmask &= ~(1 << civ)
                    if civ in allowed_mask:
                        allowed_mask &= ~(1 << civ)
                        send_net_msg(0x2A, ALL, allowed_mask, ...)
                    purged++

        // Handle reported disconnects from network layer
        if global_disconnect_counter != 0:
            saved_disconnects = copy(disconnect_table)
            reset_network_state()
            for slot = 1..6:
                if saved_disconnects[slot] > 0 AND slot.active:
                    purged++
                    civ = slot.civ
                    log_player_name(slot)
                    show_msg(0x53, ...)   // "player disconnected"
                    clear_timeout_flag(civ)
                    human_civs_bitmask &= ~(1 << civ)
                    disconnect_socket(slot.connection_id)
                    update_allowed_mask(civ)
                    show_dialog("LOSTCLIENT")

        // Rebuild player count
        if purged > 0:
            mp_player_slot_count = count(active slots)
            send_net_msg(0x04, ALL, ...)
            send_full_game_state(with_visibility)

    return false  // not fatal
