```
function mp_choose_players_dialog(numPlayers):
    human_civs_bitmask = 0
    firstCiv = 0
    for playerIdx = 0 to numPlayers-1:
        // Build civ selection list
        reset_dialog(1)
        build_label(playerIdx + 0x25E)  // "Player N"
        foundFirst = false
        for civSlot = 1..7:
            if civ_is_alive(civSlot):
                add_list_item: civName, leaderName, cities, units
                entryIndex = add_radio_entry(civSlot)

        current_player = chosenCiv

        show_list_dialog("GENDER", 1)
```

```
function mp_register_network_player(packet, socketId):
    ASSERT(net_is_server)  // "gNetMgr.bServer"
    savedPlayer = current_player
    savedCiv5 = activeCiv5
    savedCiv3 = activeCiv3

    if gameNotStarted AND noActivePlayers AND gameMode < 2:
        // === First join path (game not started) ===
        civIndex     = packet[+0x10]
        leaderGraphic = packet[+0x14]
        gender       = packet[+0x18]
        activeCiv3   = civIndex
        civData[civIndex].leaderIndex = leaderGraphic
        civData[civIndex].portraitSuffix = leaderGraphic
        if gender != 0:
            portraitSuffix += 21  // female offset

        current_player = civIndex
        if gameMode < 3:
            human_civs_bitmask |= (1 << civIndex)
        else:
            net_civs_bitmask |= (1 << civIndex)

        leaderTable[leaderIndex].gender = gender
        // Resolve leader's civ name from table
        leaderCivName = lookup_civ_name(leaderTable[leaderIndex].civId)
        civName = copy(leaderCivName)
        if leaderTable[leaderIndex].civId > 0:
            negate(leaderTable[leaderIndex].civId)  // mark as taken

        // Override with packet player name
        civName = strcpy(packet + 0x20)

        // If packet has custom city names (flag at +0x1C):
        if packet[+0x1C] != 0:
            negate leaderTable secondary IDs
            for citySlot = 0..6:
                resolve city name from abs(leaderCityNames[slot])
            copy city names from packet (+0x38, +0x50)
            if packet[+0x68] != 0:
                for citySlot = 0..6:
                    copy custom city name (packet + 0x6C + slot*0x18)

        civData[civIndex].customField = packet[+0x114]

    else:
        // === Subsequent join path (game started or in progress) ===
        civIndex = packet[+0x10]
        if gameMode < 3:
            human_civs_bitmask |= (1 << civIndex)
        else:
            net_civs_bitmask |= (1 << civIndex)
        activeCiv3 = civIndex
        activeCiv5 = civIndex
        current_player = civIndex
        civName = strcpy(packet + 0x20)
        if leaderTable[leaderIndex].civId > 0:
            negate(leaderTable[leaderIndex].civId)
        gender = packet[+0x18]
        civData[civIndex].flags &= ~0x200
        if gender != 0: civData[civIndex].flags |= 0x200
        leaderTable[leaderIndex].gender = gender

    // Find matching network socket slot
    for slotIdx = 0..6:
        if net_socket_array[slotIdx].socketId == socketId: break
    if slotIdx < 7:
        civ_to_socket_map[activeCiv3] = slotIdx
        net_socket_civ[slotIdx] = activeCiv3
        net_slot_active[slotIdx] = 1
        net_slot_host[slotIdx] = (socketId == 0) ? 1 : 0
        return 1  // success
    else:
        // No free socket — undo registration
        if gameMode < 3: human_civs_bitmask &= ~(1 << civIndex)
        else:            net_civs_bitmask &= ~(1 << civIndex)
        send_reject_message(0x31, leaderGraphic, civIndex)
        return 0  // failure

    // Restore saved state
    current_player = savedPlayer
    activeCiv5 = savedCiv5
    activeCiv3 = savedCiv3
```
