    clear continent_land_tile_counts[64]
    clear continent_sea_tile_counts[64]

    tileBodyIds = allocGlobal(numTiles*4) // 2 bytes per tile (uint16 array)

                                        for all scanned tiles:

                    if currentBody == 0:

                        if !continuingPrevRow:

                            while countArray[++nextBodyId] != 0 AND nextBodyId < 0x7FFE: pass
                        currentBody = nextBodyId

                    countArray[currentBody]++
