function ai_diplomacy_turn_processing(civId):
    # Phase 1: Clear flags and manage government
    civ = civData[civId]
    civ.flags &= 0xFFB7                          # clear bits 3 and 6

    if civ.government == ANARCHY:                 # civ+0x15 == 0
        if isHuman(civId) OR (NOT isHuman(civId) AND game.flags & 3 != 0):
            if game.flags & 3 == 0 OR has_tech(civId, COMMUNISM):
                set_government(civId, 1)          # auto-select government
                if isHuman(civId):
                    civ.flags &= 0xFFFE           # clear senate bit
        else if NOT isHuman(civId):
            if civ.flags & 1 == 0:
                set_government(civId, 1)
            else:
                civ.flags &= 0xFFFE

    # If not human AND has senate flag set, restore government
    if NOT isHuman(civId) AND civ.flags & 1:
        set_government(civId, civ.government)

    if civId == 0: return                         # barbarians skip diplomacy

    # Phase 2: Roll AI random seed and toggle senate override
    civ.aiRandomSeed = rand() % 100               # civ+0x16
    if rand() % 3 == 0:
        civ.flags ^= 0x04                         # toggle senate override bit

    # Phase 3: Decrement patience (every 3rd turn)
    if turnNumber % 3 == 0 AND civ.patience > 0:  # civ+0x1F
        civ.patience -= 1

    # Phase 4: Main diplomacy loop — iterate all other civs
    for otherCiv in 1..7:
        if MP_VERSION > 2: process_messages(1, 0)

        # Save and clear alliance-violation memory
        if treaty[otherCiv][civId] & 0x40:         # bit 0x40 = prev alliance violation
            tempAllianceViolation[otherCiv] = 1
        else:
            tempAllianceViolation[otherCiv] = 0

        # Clear transient treaty flags for this pair
        treaty[otherCiv][civId] &= 0xFF5FBFFF      # clear bits 14,17,23

        if NOT isAlive(otherCiv) OR otherCiv == civId: continue
        if isHuman(civId): continue                 # AI-only processing

        # --- Alliance violation detection ---
        if treaty[otherCiv][civId] & 0x20:          # hatred/alliance violation flag
            toleranceRolls = clamp(3 - (civ.tolerance >> 2), 1, 3)
            rollResult = (toleranceRolls <= 1) ? 0 : rand() % toleranceRolls

            if rollResult == 0:
                if NOT atWar(otherCiv, civId):      # treaty & 0x08 == 0
                    # Declare war due to alliance violation
                    treaty[otherCiv][civId] &= 0xFFFFFFD9  # clear ceasefire+peace
                    treaty[otherCiv][civId] |= 0x80840     # set war flags
                    contactTurn[civId][otherCiv] = max(contactTurn, turn - 8)
                else:
                    # Already at war: set attitude to max hostility
                    set_attitude(civId, otherCiv, 100)
                    contactTurn[civId][otherCiv] = 0xFFFF
                    treaty[otherCiv][civId] &= ~0x20       # clear violation flag

        # --- Periodic diplomacy actions (for AI-AI and AI-human) ---
        if NOT isHuman(otherCiv):
            # Clear some periodic flags
            treaty[otherCiv][civId] &= ~0x40000     # clear bit 18

            if turnNumber & 0x1F == 0:              # every 32 turns
                if NOT (treaty[otherCiv+1][civId] & 0x08):
                    treaty[otherCiv][civId] &= ~0x800
                treaty[otherCiv][civId] &= ~0x80000

            if turnNumber & 0x0F == 0:              # every 16 turns
                if NOT atWar(otherCiv, civId) AND NOT isEmbassy(civId, otherCiv):
                    if NOT ceasefire AND NOT (ceasefire AND game.flags & 1 == 0):
                        if haveContact AND hasAlliance(otherCiv, civId)
                           AND NOT isEmbassy(civId, otherCiv):
                            # Check if war should end ("WARENDS" event)
                            # Complex visibility check: embassy OR Navigation OR Writing
                            if canSeeWarEnd(civId, otherCiv):
                                fire_event("WARENDS", civId, otherCiv)
                            # Notify human players in MP
                            if MP_VERSION > 2:
                                for humanCiv in 1..7:
                                    if isHuman(humanCiv) AND canSee(humanCiv):
                                        send_mp_notification(0x50, ...)
                    else if ceasefire:
                        # Ceasefire: attempt diplomacy encounter
                        diplomacy_encounter(civId, otherCiv, 0, 0)
                    else:
                        # At war: attempt tech negotiation
                        tech_negotiation(civId, otherCiv)

                # Post-diplomacy: check shared visibility
                if haveContact AND NOT isEmbassy:
                    unknown_diplo_check(civId, otherCiv, 1)

                treaty[otherCiv][civId] &= ~0x400   # clear bit 10

            else if turnNumber & 0x07 == 0:
                # Every 8 turns: tech negotiation if at war
                if atWar AND NOT isEmbassy:
                    tech_negotiation(civId, otherCiv)

    # Phase 5: AI-to-human diplomacy evaluation
    if NOT isHuman(civId):
        for otherCiv in 1..7:
            if MP_VERSION > 2: process_messages(1, 0)
            if isHuman(otherCiv) AND NOT gamePaused AND haveContact:
                if otherCiv == currentHumanCiv:
                    evaluate_diplomacy_toward_human(civId, otherCiv)
                else if MP_VERSION > 2:
                    send_mp_message(0xA0, ...)      # delegate to remote

        # Phase 6: Unknown diplomacy post-processing
        unknown_diplo_post(civId)                    # FUN_0055f7d1

        # Phase 7: Alliance proposal loop
        for otherCiv in 1..7:
            if MP_VERSION > 2: process_messages(1, 0)
            if isHuman(otherCiv):
                if otherCiv == currentHumanCiv:
                    propose_alliance_or_crusade(civId, otherCiv)
                else if MP_VERSION > 2:
                    send_mp_message(0xA1, ...)

function ai_evaluate_diplomacy_toward_human(aiCiv, humanCiv):
    # Phase 1: Get border information
    get_border_info(aiCiv, humanCiv)
    # Sets: DAT_006ab5e4 = nearestCityIdx, DAT_006ab5e8 = borderScore,
    #        DAT_006ab5ec = intruderCount, DAT_00633ac8 = intruderDetailCount

    nearestCity = DAT_006ab5e4
    intruderCount = DAT_006ab5ec

    # Phase 2: Handle diplomatic events (if intruder detail > 0 and not at war/alliance)
    if intruderDetailCount != 0 AND NOT (treaty[humanCiv][aiCiv] & 0x2008):
        cityX = city[nearestCity].x
        cityY = city[nearestCity].y

        if NOT ceasefire(humanCiv, aiCiv):              # treaty & 0x04 == 0
            if tempAllianceViolation[humanCiv] == 0:
                # Peaceful: check for admiration or near-city event
                attitudeBoost = borderScore
                # Random admiration chance
                randomBit = rand() & 1

                if treaty[humanCiv][aiCiv] & 0x10:      # trespass flag
                    intruderCount += 1

                if tolerance + intruderCount > 2:
                    rollMax = tolerance + intruderCount
                    roll = (rollMax <= 1) ? 0 : rand() % rollMax
                    if roll > 1:
                        randomBit = 0
                        set_treaty_flags(aiCiv, humanCiv, 0x20)  # alliance violation

                if (turn + aiCiv) & 3 == 0:
                    hasNearCity = check_near_city(humanCiv, aiCiv, cityX, cityY)
                    if hasNearCity:
                        if NOT already_visible(nearestCity, humanCiv):
                            reveal_city(nearestCity, humanCiv)
                            update_map(cityX, cityY)
                        if randomBit == 0:
                            fire_event("NEARCITY", ...)
                        else:
                            fire_event("ADMIRECITY", ...)
                    cleanup_border_check()
        else:
            # Ceasefire active
            if tempAllianceViolation[humanCiv] != 0:
                # Terms reminder
                fire_event("TERMS", ...)
                goto attitude_scoring

            # Check contact cooldown (within 3 turns of contact)
            if contactTurn[aiCiv][humanCiv] < turn AND turn - contactTurn < 3:
                goto attitude_scoring

            if intruderCount == 0:
                # No intruders: demand tribute + check for intrusion
                demand_tribute(humanCiv, aiCiv)
                unknown_check(humanCiv, aiCiv)
                if NOT already_visible: reveal_city + set_visibility + update_map
                attitudeBoost = borderScore - 1
                if intruderDetailCount < 2:
                    fire_event("INTRUDER", ...)
                else:
                    fire_event("INTRUDERS", ...)
                goto attitude_scoring
            else:
                # Intruders present during ceasefire = violation
                demand_tribute(humanCiv, aiCiv)
                unknown_check(humanCiv, aiCiv)
                reveal_city_if_needed()
                if intruderDetailCount < 2:
                    response = fire_event("VIOLATOR", ...)
                else:
                    response = fire_event("VIOLATORS", ...)

                if response == 0:                   # player chose to withdraw
                    # Withdraw all units from area
                    for each unit:
                        if unit is alive AND unit has ceasefire flag
                           AND unit.owner == humanCiv:
                            clear_ceasefire_flag(unit)
                            find_nearest_friendly_city -> move unit there
                            clear_orders, clear_home_city
                    update_display()
                    fire_event("WITHDRAWN", ...)
                else:                               # player chose to violate
                    # Senate scandal check (Democracy/Republic)
                    if humanCiv.government == DEMOCRACY:
                        goto senate_scandal
                    elif rand() & 1 AND humanCiv.government == REPUBLIC:
                        goto senate_scandal
                    # senate_scandal:
                    if NOT (game.scenarioFlags & 0x80) OR NOT (scenarioBits & 1):
                        show_message("SENATESCANDAL")
                        set_government(humanCiv, ANARCHY)

                    set_attitude(aiCiv, humanCiv, 50)   # 0x32 = hostile
                    declare_war(humanCiv, aiCiv, -1)
                    fire_event("VIOLATE", ...)
                cleanup_border_check()

    # Phase 3: Clear ceasefire flags on all units
    for each unit:
        if unit is alive:
            unit.flags &= ~0x04                     # clear ceasefire marker

    # Phase 4: Attitude scoring algorithm
    attitude_scoring:
    warCount = 0
    techRankCount = 0       # how many civs have more tech than humanCiv
    allianceStrength = 0    # strength of humanCiv's allies

    for otherCiv in 1..7:
        if otherCiv.techCount > humanCiv.techCount:
            techRankCount += 1
        if count_wars_between(aiCiv, otherCiv) != 0:
            warCount += 1
        if isAllied(otherCiv, humanCiv):            # treaty & 0x2000
            allianceStrength += 1
            if humanCiv.militaryPower < otherCiv.militaryPower:
                allianceStrength += 1               # ally is stronger

    # Score modifiers when at war
    if atWar(humanCiv, aiCiv):
        if warCount == 0:                           # no other wars
            if aiCiv.gold < humanCiv.gold: score += 1
            if aiCiv.techCount < humanCiv.techCount AND difficulty > 0: score += 1
            if humanCiv.techCount < aiCiv.techCount: score -= 1
            if aiCiv.militaryPower < humanCiv.militaryPower: score += 1
            if humanCiv.militaryPower < aiCiv.militaryPower: score -= 1
        else:
            # Multiple wars: penalize based on expansionism
            if leader.expansionism < 1 OR warCount > 1:
                penalty = warCount - leader.expansionism - 1
                if penalty < 2: penalty = 1
                score -= penalty

        if aiCiv.techCount + 8 < humanCiv.techCount: score += 1

    # Ceasefire penalty
    if ceasefire(humanCiv, aiCiv): score -= 2

    # Alliance bonus
    if NOT isAllied(humanCiv, aiCiv): score -= 1

    # Spaceship penalty (late game)
    if humanCiv.persona < 7 AND turn > 200:
        spaceshipPenalty = 7 - humanCiv.persona
        if check_spaceship(): spaceshipPenalty = 1
        if turn < 400: spaceshipPenalty = (spaceshipPenalty + 1) >> 1
        score -= spaceshipPenalty

    # Power ranking bonus (persona 7 = top power)
    if humanCiv.persona == 7 AND humanCiv.cityCount > 3
       AND turn > 200 AND difficulty > 0:
        powerBonus = difficulty / 3 + 1
        if tolerance - attitudes[aiCiv][humanCiv] <= allianceStrength
           AND NOT (treaty[humanCiv][aiCiv] & 0x10):
            powerBonus >>= 1
        score += powerBonus

    # Spaceship active penalty
    if humanCiv.spaceshipActive:
        score -= 1
        if aiCiv.spaceshipActive == 0 AND score > 0:
            score -= 1

    # Leader personality modifier
    personalityMod = leader.expansionism * 3 + leader.militarism * 2
    if personalityMod < -1: personalityMod = -2

    # Power differential
    powerDiff = humanCiv.persona - aiCiv.persona
    if powerDiff < 0: powerDiff /= 2
    elif NOT atWar: powerDiff /= 2

    score += personalityMod + powerDiff

    # Military power comparison (when peaceful or weak alliances)
    if NOT (game.flags & 1) OR (tolerance - attitude <= allianceStrength
       AND NOT trespass):
        if aiCiv.militaryPower * 4 < humanCiv.militaryPower: score -= 1
        if aiCiv.militaryPower * 2 < humanCiv.militaryPower: score -= 1
        if aiCiv.militaryPower * 3/2 < humanCiv.militaryPower: score -= 1

    # Not at war: bonus if AI is stronger
    if NOT atWar:
        if humanCiv.militaryPower < aiCiv.militaryPower: score += 1
        if humanCiv.militaryPower * 2 < aiCiv.militaryPower: score += 1

    # Wonder effects
    if has_tech(humanCiv, GREAT_WALL) OR has_tech(humanCiv, NAVIGATION):
        if score < 1: score -= 1
        else: score >>= 1

    # Lowest tech rank bonus
    if techRankCount == 0: score += 1

    # Tech count comparison with tolerance modifier
    if aiCiv.techCount < humanCiv.techCount:
        score += (1 - leader.tolerance)

    # Alliance floor
    if isAllied(humanCiv, aiCiv) AND score < 1:
        score = 0

    # Democracy wonder effect
    if has_tech(aiCiv, DEMOCRACY_ADVANCE):
        score += 1
    if has_tech(humanCiv, DEMOCRACY_ADVANCE):
        if score > 0: score /= 2
        score -= 1
        if score >= 0: score -= 1

    # No contact = neutral
    if NOT haveContact(humanCiv, aiCiv):
        score = 0

    # Apply final attitude
    set_attitude(aiCiv, humanCiv, score)

    # MP sync
    if MP_VERSION > 2:
        mp_sync(0xFF, 2, 0, 0, 0)
        XD_FlushSendBuffer(5000)

function ai_propose_alliance_or_crusade(aiCiv, humanCiv):
    # Timing gate: only fires when (turn & 0x1F) == aiCiv << 2
    if (turn & 0x1F) != aiCiv << 2: return

    # Attitude check: tolerance - attitude[aiCiv][humanCiv] must be < 6
    if tolerance - attitude[aiCiv][humanCiv] >= 6: return

    for targetCiv in 1..7:
        if targetCiv == aiCiv OR targetCiv == humanCiv: continue
        if NOT isAlive(targetCiv): continue

        # Prerequisite: aiCiv is at war with targetCiv (treaty & 0x02)
        if NOT (treaty[aiCiv][targetCiv] & 0x02): continue
        # humanCiv is NOT at war/alliance with targetCiv
        if treaty[aiCiv][humanCiv] & 0x2008: continue
        if count_wars(humanCiv, targetCiv) != 0: continue
        # Both aiCiv and humanCiv have contact with each other and targetCiv
        if NOT haveContact(aiCiv, humanCiv): continue
        if NOT haveContact(targetCiv, humanCiv): continue

        # Path A: "HELPME" — aiCiv is weaker, offers gold + tech for alliance
        if aiCiv.militaryPower * 2 <= targetCiv.militaryPower:
            hasNearCity = check_near_city(humanCiv, aiCiv, -1, -1)
            if hasNearCity == 0:
                # Can't reach human: just bump contact cooldown
                contactTurn[aiCiv][humanCiv] = max(contactTurn, turn - 15)
            else:
                # Calculate gold offer: (aiCiv.gold / 100) clamped to [1,10]
                goldOffer = clamp(aiCiv.gold / 100, 1, 10)

                # Find best tech to offer
                bestTech = -1
                bestScore = -99
                for tech in 0..99:
                    if tech.prereq1 == -2 AND tech.prereq2 == -2: continue
                    if humanCiv.has_tech(tech): continue
                    if aiCiv.has_tech(tech) OR
                       (targetCiv.has_tech(tech) AND canShareTech):
                        techScore = score_tech(humanCiv, tech)
                        if techScore > bestScore:
                            bestTech = tech
                            bestScore = techScore

                # Build dialog string
                dialogKey = "HELPME"
                if atWar(targetCiv, humanCiv):     # already at war with target
                    dialogKey += "INSTEAD"
                if bestTech >= 0:
                    dialogKey += tech_offer_suffix
                    set_tech_display(2, techName[bestTech])

                set_string_param(1, civName[targetCiv])
                set_gold_display(0, goldOffer * 50)

                response = fire_event(dialogKey, ...)

                if response == 0:                   # human declined
                    contactTurn[aiCiv][humanCiv] = max(contactTurn, turn - 14)
                else:                               # human accepted
                    # Transfer tech
                    if bestTech >= 0:
                        if NOT aiCiv.has_tech(bestTech):
                            give_tech(aiCiv, bestTech, targetCiv, 0, 0)
                        give_tech(humanCiv, bestTech, aiCiv, 0, 0)

                    # Set alliance treaty flags
                    set_treaty_flags(aiCiv, humanCiv, 0x0C)   # ceasefire+peace
                    declare_war(humanCiv, targetCiv, aiCiv)
                    set_treaty_flags(humanCiv, targetCiv, 0x2000)  # alliance
                    treaty[targetCiv][humanCiv] |= 0x80800     # war declaration
                    contactTurn[targetCiv][humanCiv] = turn
                    contactTurn[aiCiv][humanCiv] = turn + 16

                    # Transfer gold
                    aiCiv.gold -= goldOffer * 50
                    humanCiv.gold += goldOffer * 50
                    refresh_status_panel(1)

            cleanup()
            return

        # Path B: "CRUSADE"/"JIHAD" — aiCiv is stronger, proposes joint attack
        if aiCiv.militaryPower < targetCiv.militaryPower
           AND (humanCiv.persona < 6 OR aiCiv.persona < 6):
            hasNearCity = check_near_city(humanCiv, aiCiv, -1, -1)
            if hasNearCity == 0:
                contactTurn[aiCiv][humanCiv] = max(contactTurn, turn - 15)
            else:
                set_string_param(1, civName[targetCiv])
                # Choose "CRUSADE" or "JIHAD" based on existing war status
                if atWar(aiCiv, humanCiv):
                    dialogKey = "CRUSADE"
                else:
                    dialogKey = "CRUSADE"
                if atWar(targetCiv, humanCiv):
                    dialogKey = "JIHAD"

                response = fire_event(dialogKey, ...)

                if response == 0:                   # declined
                    contactTurn[aiCiv][humanCiv] = max(contactTurn, turn - 14)
                else:                               # accepted
                    set_treaty_flags(aiCiv, humanCiv, 0x0C)
                    declare_war(humanCiv, targetCiv, aiCiv)
                    contactTurn[aiCiv][humanCiv] = turn + 16
                    set_treaty_flags(humanCiv, targetCiv, 0x2000)
                    treaty[targetCiv][humanCiv] |= 0x80800
                    contactTurn[targetCiv][humanCiv] = turn

            cleanup()
            return

        # Clear "attacked" flag for this pair
        treaty[aiCiv][targetCiv] &= ~0x200

        # MP sync
        if MP_VERSION > 2:
            mp_sync(0xFF, 2, 0, 0, 0)
            XD_FlushSendBuffer(5000)

function screen_reveal_animation(surface, rect, blockSize, mode):
    # Animated screen transition effect (wipe/dissolve)
    # mode: 1=bottom-to-top, 2/5=top-to-bottom, 3=left-to-right, 4=right-to-left, 5=random
    if rect == NULL: rect = full screen

    cols = rect.width / blockSize
    rows = rect.height / blockSize
    totalBlocks = rows * cols

    blocks = GlobalAlloc(totalBlocks * 8)           # x,y pairs

    # Fill block array in order based on mode
    switch mode:
        case 1: bottom-to-top row scan
        case 2,5: top-to-bottom row scan
        case 3: left-to-right column scan
        case 4: right-to-left column scan

    # Reveal blocks one at a time
    if mode == 5:
        # Random order: pick random block, swap with last, shrink pool
        while totalBlocks > 0:
            idx = random(0, totalBlocks - 1)
            reveal_block(blocks[idx])
            swap blocks[idx] with blocks[last]
            totalBlocks -= 1
    else:
        # Semi-random: pick from recent window
        while totalBlocks > 0:
            windowStart = max(0, totalBlocks - 1 - cols*20/blockSize)
            idx = random(windowStart, totalBlocks - 1)
            reveal_block(blocks[idx])
            swap and shrink

    GlobalFree(blocks)
