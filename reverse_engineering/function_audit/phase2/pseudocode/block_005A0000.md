## 1. UI.Popup -- Dialog System

The popup system is a custom listbox/menu/prompt UI framework centered around a large struct
(`this` via `in_ECX`, globally `DAT_006cec84`). It manages item lists, scrollbars, keyboard
navigation, multi-column layouts, tab strips, and text rendering. The `@` directive parser
(`popup_parse_text_file`) is the engine behind all in-game popups, menus, and prompts.

```
function popup_draw_text_truncated(text, x, y, use_alt_color, highlight):
    column_width = this.columnWidths[this.currentPane]
    while text[0] != '\0':
        if measure_text(text) <= column_width - 6: break
        text[strlen(text) - 1] = '\0'    // truncate by 1 char until fits

    if use_alt_color == 0:
        fg_color = this.fgColor          // +0x94
        shadow_color = this.shadowColor  // +0x98
    else:
        fg_color = this.altFgColor       // +0xA0
        shadow_color = this.altShadowColor // +0xA4

    if shadow_color != fg_color:  // draw shadow offset (+2,+1)
        set_color(shadow_color)
        draw_text(this.surface, text, x+2, y+1, 5)
        set_color(fg_color)
        draw_text(this.surface, text, x+1, y, 5)

    if highlight:
        set_color(this.highlightColor)    // +0x6C
    else:
        set_color(fg_color)
    draw_text(this.surface, text, x, y, 5)
```

function popup_draw_split_text(text, y_offset, p3, p4, p5):
    right_part = NULL
    pipe = strchr(text, '|')
    if pipe != NULL:
        right_part = pipe + 1
        *pipe = '\0'

    // Draw left-justified part
    x = this.paneOrigins[this.currentPane]
    popup_draw_text_truncated(text, x + y_offset, p3, p4, p5)

    if right_part != NULL:
        // Draw right-justified part
        col_width = this.columnWidths[this.currentPane]
        text_width = measure_text(right_part)
        popup_draw_text_truncated(right_part, col_width + y_offset - 4 - text_width, p3, p4, p5)
        *pipe = '|'  // restore original string
```

### FUN_005a120b -- popup_draw_item

function popup_draw_item(item_node, do_end_paint):
    if item_node == NULL OR this.itemCount == 0: return 0

    pane = item_node.pane                        // +0x0C
    scroll_start = list_index(this.scrollItems[pane])
    item_index = list_index(item_node)
    flat_offset = item_index - scroll_start

    if flat_offset < 0 OR flat_offset >= this.itemsPerCol[pane] * this.columns: return 0

    column = flat_offset / this.itemsPerCol[pane]
    row = flat_offset % this.itemsPerCol[pane]
    draw_x = this.paneColX[pane] * column + this.paneX[pane]
    draw_y = this.paneRowH[pane] * row + this.paneY[pane]
    item_rect = make_rect(draw_x, draw_y, paneColX, paneRowH)

    is_selected = (this.selectedItem == item_node)
    bg_color = is_selected ? this.selBgColor : this.normalBgColor

    // custom draw callback OR standard fill + text
    if this.customDrawFn == 0 OR customDrawFn(...) == 0:
        if is_selected:
            draw_3d_border(this.surface, item_rect, light, dark)
            inset_rect(item_rect, 1, 1)
        fill_rect(this.surface, item_rect, bg_color)
        popup_draw_split_text(item_node.text, draw_x+2, draw_y, is_selected, item_node.flags & 1)

    if do_end_paint: invalidate_region(item_rect)
    return is_selected
```

function popup_redraw_visible_items():
    saved_pane = this.currentPane
    for pane_id = 0 to 1:
        if pane_id == 1 AND NOT (this.flags_3E & 0x40): continue  // skip pane 1 if no dual-pane
        this.currentPane = pane_id
        this.selectedVisible[pane_id] = 0
        pane_rect = this.paneClipRects[pane_id]

        if this.columns > 1 OR this.itemCount == 0 OR this.customDrawFn != 0:
            fill_rect(this.surface, this.paneBgRects[pane_id], this.normalBgColor)
        draw_border(this.surface, this.paneBorderRects[pane_id], this.borderDarkColor)
        set_clip(pane_rect)

        if this.itemCount != 0:
            node = this.itemListHead             // +0x228
            scroll_start = list_index(this.scrollItems[pane_id])
            pane_item_count = 0
            for i = 0 to this.itemCount - 1:
                if node == NULL: break
                if node.pane == pane_id:
                    if pane_item_count >= scroll_start AND
                       pane_item_count < scroll_start + this.itemsPerCol[pane_id] * this.columns:
                        is_sel = popup_draw_item(node, 0)
                        this.selectedVisible[pane_id] |= is_sel
                    pane_item_count++
                node = node.next

        reset_clip()
        invalidate_region(this.paneBorderRects[pane_id])
    this.currentPane = saved_pane
```

function popup_select_item(item_node, redraw):
    if item_node == 0 OR this.selectedItem == item_node: return
    old_selected = this.selectedItem

    scroll_start = list_index(this.scrollItems[this.currentPane])
    item_idx = list_index(item_node)
    scrolled = false

    if item_idx < scroll_start:
        scrolled = true
        scroll_start = item_idx

    // advance scroll until item is visible
    while item_idx >= scroll_start + this.itemsPerCol[pane] * this.columns:
        if this.columns == 1: scroll_start++
        else: scroll_start += this.itemsPerCol[pane]
        scrolled = true

    // clamp scroll position
    max_scroll = popup_align_to_column(this.totalItems - 1)
    scroll_start = clamp(scroll_start, 0, max_scroll)
    this.scrollItems[pane] = list_node_at(scroll_start)
    this.selectedItem = item_node

    // update scrollbar position
    if this.vscroll_pos[pane]: set_scrollbar_pos(...)
    if this.vscroll_page[pane]: set_scrollbar_page(...)

    if redraw:
        if scrolled:
            popup_redraw_visible_items()
        else:
            popup_draw_item(old_selected, 1)  // unhighlight old
            popup_draw_item(item_node, 1)     // highlight new
        if this.changeCallback: this.changeCallback(this, item_node.value)
```

function popup_hittest(x, y):
    popup_select_pane_by_coords(x, y)
    pane = this.currentPane
    pane_rect = this.paneHitRects[pane]    // +0x158..+0x164

    if this.hscroll[pane] != 0:
        if x < pane_rect.left: return -3   // left of pane
        if x >= pane_rect.right: return -4  // right of pane

    if y < pane_rect.top: return -1         // above pane
    if y >= pane_rect.bottom: return -2     // below pane
    if x < pane_rect.left: return -3
    if x >= pane_rect.right: return -4

    scroll_start = list_index(this.scrollItems[pane])
    col = (x - pane_rect.left) / this.columnWidths[pane]
    row = (y - pane_rect.top) / this.rowHeights[pane]
    flat_index = col * this.itemsPerCol[pane] + row + scroll_start
    return clamp(flat_index, 0, this.totalItems - 1)
```

function popup_on_keypress(vkey):
    if popup == 0: return

    // --- TAB MODE KEYS (flags_3E & 2) ---
    if popup has tab mode:
        switch vkey:
            case DOWN/0xC1: advance to next tab, wrap around, repaint
            case UP/0xC0: go to previous tab, repaint
            case ENTER/0xD1: confirm current tab via popup_on_tab_click
        return

    if NOT list_mode (flags & 0x41000 != 0x1000): return
    if popup.itemCount == 0: vkey = 0  // no items, skip navigation

    switch vkey:
        case END/0xC6:
            if single-column: jump to last item
            else: remap to RIGHT, re-enter switch
        case HOME/0xC4:
            if current > 0: jump to first item, redraw
        case DOWN/0xC1:
            // walk linked list forward to next item in same pane
            node = selected.next_in_pane
            if found AND different from selected: select_item(node, 1)
        case UP/0xC0:
            // walk linked list backward to previous item in same pane
            node = selected.prev_in_pane
            if found: select_item(node, 1)
        case PAGEDOWN/0xC3:
            if multi-column: jump forward by itemsPerCol
            else: remap to DOWN
        case PAGEUP/0xC2:
            if multi-column: jump backward by itemsPerCol
            else: remap to UP
        case RIGHT/0xA3:
            if single-column: jump forward by itemsPerCol (page)
            else: remap to PAGEDOWN
        case LEFT/0xA9:
            if single-column: jump backward by itemsPerCol
            else: remap to PAGEUP
```

## Appendix: Game-State Logic in PBEM/Network Code

The following game-state mutations were identified in the PBEM and multiplayer functions:

1. **pbem_game_setup (FUN_005ae580)**: Sets `DAT_00655b02 = 2` (save format version),
   randomizes 5 map generation parameters, initializes per-civ email arrays and turn
   notification flags, and dispatches to map generation / game initialization.

2. **wait_production_server (FUN_005aa0e5)**: Modifies `DAT_00628048` (hot-seat civ),
   `DAT_00628044` (game_active flag), `DAT_00655b0b` (human civ bitmask), `DAT_006ced20`
   (per-civ ready flags), `DAT_00654fa8` (fast-forward mode), `DAT_00633a78` (auto-play
   timer). These are multiplayer synchronization state -- not core game model.

3. **wait_production_client (FUN_005ab2d5)**: Same globals plus reconnection state
   (`DAT_006ad698`, `DAT_006ad685`). Also multiplayer sync, not game model.

4. **editor_load_unit_types_from_buffer (FUN_005af140)**: Directly writes to the unit
   type array (`DAT_0064b1b8`), modifying game rules at runtime. This is cheat/scenario
   editor functionality -- not normal gameplay.

None of these contain game logic that needs porting for the JS engine. The PBEM setup
flow's map-gen parameter randomization is already handled by `generateMap()` in
`engine/mapgen.js`.
